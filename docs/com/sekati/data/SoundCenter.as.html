     9  import com.sekati.core.FWDepth;
    10  import com.sekati.math.MathBase;
    11  import com.sekati.utils.ClassUtils;
    12  
    13  /**
    14   * centralizes & solves issues calling linked sounds from dll's or locally
    15   * extrapolated on ca.nectere.dllFactory.SoundFactory
    16   */
    17  class com.sekati.data.SoundCenter {
    18  
    19  	private static var _instance:SoundCenter;
    20  	private var _holder:MovieClip;
    21  	private var _gSound:Sound;
    22  	private var _gMute:Boolean;
    23  	private var _gVolume:Number;
    24  	private var _gPan:Number;
    25  
    26  	/**
    27  	 * Singleton Private Constructor
    28  	 */
    29  	private function SoundCenter () {
    30  		_holder = ClassUtils.createEmptyMovieClip(com.sekati.display.BaseClip, _level0, "__SoundCenter__",{_depth: FWDepth.SoundCenter});
    31  		//no param means it'll affect the whole application
    32  		_gSound = new Sound ();
    33  		_gMute = false;
    34  	}
    35  	
    36  	/**
    37  	 * Singleton Accessor
    38  	 * @return SoundManager
    39  	 */
    40  	public static function getInstance ():SoundCenter {
    41  		if (!_instance) {
    42  			_instance = new SoundCenter ();
    43  		}
    44  		return _instance;
    45  	}
    46  
    47  	/**
    48  	 * shorthand singleton accessor getter
    49  	 */
    50  	 public static function get $():SoundCenter {
    51  	 	return SoundCenter.getInstance();	
    52  	 }	
    53  
    54  	/**
    55  	 * returns the holder clip so we can load the DLL swf in
    56  	 * @return MovieClip
    57  	 */
    58  	public function getHolder ():MovieClip {
    59  		return _holder;
    60  	}
    61  
    62  	/**
    63  	 * returns the global volume setting
    64  	 * @return Number
    65  	 */
    66  	public function getVolume ():Number {
    67  		return _gVolume;
    68  	}
    69  
    70  	/**
    71  	 * sets the global volume
    72  	 * @param n (Number) 0 to 100
    73  	 * @return Void
    74  	 */
    75  	public function setVolume (n:Number):Void {
    76  		_gVolume = MathBase.constrain (n, 0, 100);
    77  		_gSound.setVolume (_gVolume);
    78  	}
    79  
    80  	/**
    81  	 * returns the global pan settings
    82  	 * @return Number
    83  	 */
    84  	public function getPan ():Number {
    85  		return _gPan;
    86  	}
    87  
    88  	/**
    89  	 * sets the global pan
    90  	 * @param n (Number) -100 to 100
    91  	 */
    92  	public function setPan (n:Number):Void {
    93  		_gPan = MathBase.constrain (n, -100, 100);
    94  		_gSound.setPan (_gPan);
    95  	}
    96  
    97  	/**
    98  	 * play a sound
    99  	 * @param linkage (String) linkage id for sound
   100  	 * @param loops (Number) number of times to play sound
   101  	 * @param msOffset (Number) miliseconds offset
   102  	 * @return Void
   103  	 */
   104  	public function play (linkage:String, loops:Number, msOffset:Number):Void {
   105  		if (loops == null) {
   106  			loops = 0;
   107  		}
   108  		if (msOffset == null) {
   109  			msOffset = 0;
   110  		}
   111  		var s:Sound = new Sound (_holder);
   112  		s.attachSound (linkage);
   113  		s.start (msOffset / 1000, loops);
   114  	}
   115  
   116  	/**
   117  	 * toggle mute global application audio
   118  	 * @param isMute (Boolean) optional force state
   119  	 * @return Void
   120  	 */
   121  	public function mute (isMute:Boolean):Void {
   122  		if (!_gMute || isMute) {
   123  			_gSound.setVolume (0);
   124  			_gMute = true;
   125  		} else if (_gMute || !isMute) {
   126  			_gSound.setVolume (_gVolume);
   127  			_gMute = false;
   128  		}
   129  	}
   130  }
