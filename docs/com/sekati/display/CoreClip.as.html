     8  import com.sekati.display.ICoreClip;
     9  import com.sekati.core.KeyFactory;
    10  import com.sekati.utils.MovieClipUtils;
    11  /**
    12   * Base class mix-in that all MovieClip UI subclasses should extend
    13   * instead of MovieClip for standardized initialization & deconstruction.
    14   * @see {@link com.sekati.display.AbstractClip}
    15   */
    16  class com.sekati.display.CoreClip extends MovieClip implements ICoreClip {
    17  	private var _this:MovieClip;
    18  	/**
    19  	 * private constructor; class is initialized via the MovieClip.onLoad event.
    20  	 */
    21  	private function CoreClip() {}
    22  	/**
    23  	 * onLoad does core setup when clip registers on stage via onLoad. 
    24  	 * Do not override this in subclasses; instead override configUI.
    25  	 * @return Void
    26  	 */
    27  	public function onLoad():Void {
    28  		_this = this;
    29  		_this.cacheAsBitmap = true;
    30  		KeyFactory.inject(_this);
    31  		configUI();
    32  	}
    33  	/**
    34  	 * Configure UI and initialize behavior; should be overwritten by subclasses.
    35  	 */
    36  	public function configUI():Void {}
    37  	/**
    38  	 * Destroy object elements and events for proper garbage collection.
    39  	 * This is a generic destroy method to insure that, at a minimum, the 
    40  	 * enterFrame and clip itself are removed when destroy is called. 
    41  	 * This behavior can be overwritten by CoreClip's subclasses. 
    42  	 * If you wish to use CoreClip's destroy in addition to your subclass 
    43  	 * destroy you may do so via: 
    44  	 * {@code 
    45  	 * 	super.destroy(); 
    46  	 * }
    47  	 */
    48  	public function destroy():Void {
    49  		_this.onEnterFrame = null;
    50  		MovieClipUtils.rmClip(_this);
    51  	}
    52  	/**
    53  	 * give verbose trace output
    54  	 * @return String
    55  	 */
    56  	public function toString():String {
    57  		var str:String = _this._name + "= { \n";
    58  		for (var i in _this) {
    59  			str += "\t" + i + ": " + _this[i].toString() + "\n";
    60  		}
    61  		str += "};";
    62  		return str;
    63  	}
    64  	//
    65  }
