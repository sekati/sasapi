    11  /**

    14  class com.sekati.display.CoreClip extends MovieClip implements ICoreClip {
    15  	private var _this:MovieClip;
    16  	/**
    17  	 * private constructor as class is initialized via the MovieClip.onLoad event
    18  	 */
    19  	private function CoreClip() {}
    20  	/**
    21  	 * 
    22  	 * @return Void
    23  	 */
    24  	public function onLoad():Void {
    25  		_this = this;
    26  		_this.cacheAsBitmap = true;
    27  		RUID.key(_this);
    28  		configUI();
    29  	}
    30  	/**
    31  	 * configure UI and initialize behavior - to be overwritten by subclass
    32  	 */
    33  	public function configUI():Void {}
    34  	/**
    35  	 * deconstruct object elements and events for proper garbage collection
    36  	 */
    37  	public function destroy():Void {
    38  		_this.onEnterFrame = null;
    39  	}
    40  	/**
    41  	 * give verbose trace output
    42  	 * @return String
    43  	 */
    44  	public function toString():String {
    45  		var str:String = _this._name + "= { \n";
    46  		for (var i in _this) {
    47  			str += "\t" + i + ": " + _this[i].toString() + "\n";
    48  		}
    49  		str += "};";
    50  		return str;
    51  	}

    53  }
