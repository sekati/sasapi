    57  	/**
    58  	 * App bootstrap props
    59  	 * _bootstrapChain (Array) list of methods to run to bootstrap App
    60  	 *  _bootstrapCounter (Number) bootstrap stage counter
    61  	 */
    62  	private static var _bootstrapChain:Array = ["loadConfig", "loadStyle"];
    63  	private static var _bootstrapCounter:Number = -1;

    65  	/**
    66  	 * init begins the bootstrap process
    67  	 * @return Void
    68  	 */
    69  	public static function init ():Void {
    70  		trace ("App init ...");
    71  		App.bc.broadcast ("onBootstrap");
    72  		App.bootstrap ();
    73  	}
    74  	
    75  	/**
    76  	 *loop through bootstrapChain method array
    77  	 * @return Void
    78  	 */
    79  	private static function bootstrap ():Void {
    80  		App._bootstrapCounter++;
    81  		if (_bootstrapCounter < _bootstrapChain.length) {
    82  			var methodName:String = App._bootstrapChain[App._bootstrapCounter];
    83  			App[methodName] ();
    84  		} else {
    85  			App.bc.broadcast ("onAppConfigured");
    86  			App.log.status("App","@@@ onAppConfigured EVENT FIRED - Application Full INIT!");
    87  		}
    88  	}
    89  	
    90  	/**
    91  	 * a method in the bootstrap chain failed - make retry attempts at each phase.
    92  	 * @return Void
    93  	 */
    94  	private static function bootstrap_retry():Void {
    95  		if (App.BOOTSTRAP_RETRY < App.BOOTSTRAP_RETRY_MAX) {
    96  				trace ("!!! BOOTSTRAP FAILED ON "+_bootstrapChain[_bootstrapCounter]+"() !!! ::: BOOTSTRAP RETRY ATTEMPT: ["+App.BOOTSTRAP_RETRY+"/"+App.BOOTSTRAP_RETRY_MAX+"]");
    97  				App.BOOTSTRAP_RETRY++;
    98  				App[_bootstrapChain[_bootstrapCounter]] ();
    99  		} else {
   100  			trace ("!!! BOOTSTRAP FATALLY BOMBED WITH  "+_bootstrapChain+"() !!! ::: BOOTSTRAP RETRY ATTEMPT: ["+App.BOOTSTRAP_RETRY+"/"+App.BOOTSTRAP_RETRY_MAX+"] ::: Sorry, I tried but the application failed to boot.");
   101  		}
   102  	}

   104  	/**
   105  	 * loads and parses config.xml then broadcasts "onConfig"
   106  	 * @return Void
   107  	 */
   108  	private static function loadConfig ():Void {
   109  		var oXML:XML = new XML ();
   110  		var o:Object = new Object ();
   111  		oXML.ignoreWhite = true;
   112  		var xmlLoaded:Function = function (success:Boolean):Void {
   113  			if (success) {
   114  				o = new XML2Object ().parseXML (oXML);
   115  				// dump object data into static App vars
   116  				App.APP_NAME = o.config.attributes.name + " v" + o.config.attributes.version;
   117  				App.CROSSDOMAIN_URI = (o.config.crossdomain_uri.data != undefined) ? o.config.crossdomain_uri.data : "crossdomain.xml";
   118  				App.CROSSDOMAIN_URI = (StringValidation.isURL(App.CROSSDOMAIN_URI)) ? App.CROSSDOMAIN_URI : NetBase.getPath()+App.CROSSDOMAIN_URI;
   119  				App.CSS_URI = o.config.css_uri.data;
   120  				App.DEBUG_ENABLE = (o.config.debug_enable.data == "true") ? true : false;
   121  				App.FLINK_ENABLE = (o.config.flink_enable.data == "true") ? true : false;
   122  				App.TRACK_ENABLE = (o.config.track_enable.data == "true") ? true : false;
   123  				App.KEY_ENABLE = (o.config.key_enable.data == "true") ? true : false;
   124  				App.DATA_PATH = (o.config.data_path.data != undefined) ? o.config.data_path.data : "";
   125  				App.BUFFER_TIME = Number (o.config.buffer_time.data);
   126  				// custom content
   127  				App.COPY1 = o.config.copy_1.data;
   128  				App.COPY2 = o.config.copy_2.data;
   129  				App.COPY3 = o.config.copy_3.data;
   130  				App.IMG1 = o.config.img_1.data;
   131  				App.IMG2 = o.config.img_2.data;
   132  				App.IMG3 = o.config.img_3.data;
   133  				App.FLV_URI = o.config.flv.data;
   134  				// enact config.xml settings 
   135  				if (App.DEBUG_ENABLE == true) {
   136  					App.log = Logger.getInstance();
   137  					App.log.isIDE = true, App.log.isLC = true, App.log.isSWF = false;					
   138  					App.log.trace ("App","@@@ Debug enabled ...");
   139  				}
   140  				// enable context menu                    
   141  				App.log.trace ("@@@ Setting ContextMenu ...");
   142  				var cm:ContextualMenu = new ContextualMenu(_level0);
   143  				cm.addItem(App.APP_NAME);
   144  				
   145  				// load crossdomain policy 
   146  				App.log.trace (this, "@@@ loading crossdomain policy: " + App.CROSSDOMAIN_URI);
   147  				System.security.loadPolicyFile (App.CROSSDOMAIN_URI);
   148  				delete oXML;
   149  				delete o;
   150  				App.bc.broadcast ("onConfig");
   151  				App.bootstrap ();
   152  			} else {
   153  				bootstrap_retry();	
   154  			}
   155  		};
   156  		oXML.onLoad = xmlLoaded;
   157  		oXML.load (App.CONF_URI);
   158  	}

   160  	/**
   161  	 * load App stylesheet css from {@code CSS_URI} during {@link bootstrap} sequence.
   162  	 * @return Void
   163  	 * {@code Usage:
   164  	 * 	tf.styleSheet = App.css;
   165  	 * 	tf.htmlText = "<span class='righthead_credit'>Hello World</span>";
   166  	 * }
   167  	 */
   168  	private static function loadStyle():Void {
   169  		var _styleSheet:TextField.StyleSheet = new StyleSheet ();
   170  		_styleSheet.onLoad = function (success:Boolean):Void {
   171  			if (success) {
   172  				App.css = _styleSheet;
   173  				App.log.info("App", "$$$ - Styles loaded (App.CSS) ...");
   174  				App.bootstrap ();
   175  			} else {
   176  				bootstrap_retry();	
   177  			}
   178  		};
   179  		_styleSheet.load (App.CSS_URI);
   180  	}

   182  	/**
   183  	 * wrappers to JavaScript tracking & flink deeplinking methods
   184  	 * @param page (String) page to be tracked 
   185  	 * @return Void
   186  	 */
   187  	public static function track (page:String):Void {
   188  		if (!App.TRACK_ENABLE) {
   189  			return;
   190  		}
   191  		App.log.trace ("App", "* App.track: " + "(" + page + ")");
   192  		ExternalInterface.call ("urchinTracker", page);
   193  	}

   195  	/**
   196  	 * wrapper to getFlink JavaScript lib - returns object from deeplink anchor
   197  	 * @return Object
   198  	 */
   199  	public static function getFlink ():Object {
   200  		var j:Object = ExternalInterface.call ("getFlink");
   201  		var a:Array = j.split ("/");
   202  		var p:String = (!a[1]) ? "" : a[1];
   203  		App.log.trace ("App", "* App.getFlink (" + p + ")");
   204  		var flink:Object = {page:p};
   205  		return flink;
   206  	}
   207  	
   208  	/**
   209  	 * wrapper to setFlink JavaScript lib - sets the current deeplink anchor
   210  	 * @param page (String)
   211  	 * @return Void
   212  	 */
   213  	public static function setFlink (page:String):Void {
   214  		if (!App.FLINK_ENABLE) {
   215  			return;
   216  		}
   217  		App.log.trace ("App", "* App.setFlink (" + page + ")");
   218  		ExternalInterface.call ("setFlink", null, page);
   219  	}

   221  	/**
   222  	 * wrapper to Flink JavaScript lib - returns the entire current url
   223  	 * @return String
   224  	 */
   225  	public static function getRef ():String {
   226  		var r:Object = ExternalInterface.call ("getRef");
   227  		return String(r);
   228  	}
