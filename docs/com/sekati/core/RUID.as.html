     9  /**
    10   * Runtime Unique ID generates unique Object ID's for management during application runtime
    11   */
    12  class com.sekati.core.RUID {
    13  	private static var _key:String = "__RUID";
    14  	private static var _id:Number = 0;
    15  	/**
    16  	 * Generate a runtime unique id
    17  	 * @return (Number) RUID
    18  	 */
    19  	public static function create():Number {
    20  		return _id++;
    21  	}
    22  	/**
    23  	 * return the RUID key injected into an object - used as both getter and setter.
    24  	 * @param o (Object) Object to inject RUID key in to
    25  	 * @return (Number) Runtime Unique ID
    26  	 */
    27  	public static function key(o:Object):Number {
    28  		if (!o[_key]) {
    29  			o[_key] = RUID.create();
    30  			_global.ASSetPropFlags(o, [_key], 7, 1);
    31  		}
    32  		return o[_key];		
    33  	}
    34  	
    35  	private function RUID(){}
    36  	//
    37  }
