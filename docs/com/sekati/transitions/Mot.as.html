    10  /**
    11   * Mot - Motion Styling template support for {@link caurina.transitions.Tweener}
    12   * {@code Usage:
    13   * 	Tweener.addTween(myMc, { _x:500, _y:30, base:Mot.base0 } );
    14   * 	Tweener.addTween(myMc, { base:Mot.fadeOut } );
    15   * 	Tweener.addTween(myMc, { base:Mot.fadeTo(50), onComplete:myCallBackFn} );
    16   * }
    17   */
    18  class com.sekati.transitions.Mot {
    19  	
    20  	// easing shortcuts
    21  	public static var i:Object = { quint:"easeInQuint", quad:"easeInQuad", quart:"easeInQuart", expo:"easeInExpo" };
    22  	public static var o:Object = { quint:"easeOutQuint", quad:"easeOutQuad", quart:"easeOutQuart", expo:"easeOutExpo" };
    23  	public static var io:Object ={ quint:"easeInOutQuint", quad:"easeInOutQuad", quart:"easeInOutQuart", expo:"easeInOutExpo" };
    24  	
    25  	// timing shortcuts
    26  	public static var t:Object = {s:0.3, s2:0.5, s3:0.7, d:0.3, d2:0.5, d3:1};
    27  
    28  	// deprecated "App.mot" mc_tween2 object - kept for backwards compatability
    29  	public static var m:Object = {e:"easeInOutQuint", e2:"easeOutQuint", e3:"easeInOutQuad", e4:"easeOutQuad", e5:"easeInOutQuart", e6:"easeOutQuart", e7:"easeInOutExpo", e8:"easeOutExpo", s:0.3, s2:0.5, d:0.5};
    30  	
    31  	// color shortcuts
    32  	public static var col:Object = {b:0x000000, w:0xFFFFFF, r:0xFF0000, g:0x00FF00, b:0x0000FF, y:0xFFFF00, c:0x00FFFF, m:0xFF00FF};
    33  	
    34  	// preset templates
    35  	public static function get base0():Object { 
    36  		return { time:t.s, transition:io.quint };
    37  	}
    38  	
    39  	public static function get base1():Object {
    40  		return { time:t.s, transition:io.quad };
    41  	}
    42  	public static function get base2():Object {
    43  		return { time:t.s, transition:io.expo };
    44  	}
    45  	
    46  	public static function get fadeIn():Object {
    47  		return Mot.fadeTo(100);	
    48  	}
    49  
    50  	public static function get fadeOut():Object {
    51  		return Mot.fadeTo(0);	
    52  	}
    53  	
    54  	/**
    55  	 * Return a base tween object alpha transition
    56  	 * @param a (Number)
    57  	 * @return Object
    58  	 */
    59  	public static function fadeTo(a:Number):Object {
    60  		return { _alpha:a, time:t.s, transition:io.expo };	
    61  	}
    62  	
    63  	/**
    64  	 * Return a base tween object color transition
    65  	 * @param c (Number)
    66  	 * @return Object
    67  	 */
    68  	public static function colorTo(c:Number):Object {
    69  		return { _color:c, time:t.s2, transition:io.quint };	
    70  	}	
    71  	
    72  	/**
    73  	 * Mot Private Constructor
    74  	 */
    75  	private function Mot() {
    76  	}
    77  }
