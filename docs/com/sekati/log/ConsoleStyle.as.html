   119  	/**
   120  	 * Create an empty BaseClip - but do not inherit x,y,w,h from 'style' arg.
   121  	 * @param target (MovieClip)
   122  	 * @param style (Object) ConsoleStyle.CSS reference obj
   123  	 * @return MovieClip
   124  	 * {@code Usage:
   125  	 * 	ar e:MovieClip = ConsoleStyle.$.createClip(_root, ConsoleStyle.$.CSS.console.item.bg);
   126  	 * }
   127  	 */
   128  	public function createClip (target:MovieClip, style:Object):MovieClip {
   129  		return ClassUtils.createEmptyMovieClip (com.sekati.display.BaseClip, target, style.n);
   130  	}

   132  	/**
   133  	 * Create an empty BaseClip which inherits x,y from 'style' arg.

   143  	}

   145  	/**
   146  	 * Create a styled rectangle BaseClip
   147  	 * @param target (MovieClip)
   148  	 * @param style (Object) ConsoleStyle.CSS reference obj
   149  	 * @return MovieClip
   150  	 * {@code Usage:
   151  	 * 	var r:MovieClip = ConsoleStyle.$.createStyledRectangle (_this, ConsoleStyle.$.CSS.item.bg);
   152  	 * }
   153  	 */
   154  	public function createStyledRectangle (target:MovieClip, style:Object):MovieClip {
   155  		// create the shape container clip
   156  		var r:MovieClip = createClip(target, style);	
   157  		// sanitize style.c - if it is an array (instead of a number) it means we need to alternate colors with style.cindex
   158  		var c:Number = (style.c instanceof Array) ? alternateItemBgColor() : style.c;
   159  		//trace("drawing rect: "+style.n+": "+c);
   160  		Rectangle.draw (r, new Point(0, 0), new Point(style.w, style.h), c, style.a);
   161  		r._x = style.x;
   162  		r._y = style.y;
   163  		return r;
   164  	}
   165  	
   166  	/**
   167  	 * Create a styled triangle BaseClip
   168  	 * @param target (MovieClip)
   169  	 * @param style (Object) ConsoleStyle.CSS reference obj
   170  	 * @return MovieClip
   171  	 */
   172  	public function createStyledTriangle (target:MovieClip, style:Object):MovieClip {
   173  		var t:MovieClip = createClip(target, style);
   174  		Triangle.draw (t, new Point(style.w,style.h), new Point(0,style.h), new Point(style.w,0), style.c, style.a, 0);
   175  		t._x = style.x;
   176  		t._y = style.y;
   177  		return t;	
   178  	}

   180  	/**
   181  	 * Alternate the item bg color to create "zebra striped" item entries.
   182  	 * @return Number - item bg color
   183  	 */
   184  	private function alternateItemBgColor():Number {
   185  		CSS.item.bg.cindex = (CSS.item.bg.cindex == 1) ? 0 : 1;
   186  		//trace("alternateColor ["+CSS.item.bg.cindex+"] called: "+CSS.item.bg.c[CSS.item.bg.cindex]);
   187  		return CSS.item.bg.c[CSS.item.bg.cindex];
   188  	}

   190  	/**
   191  	 * Create a styled textfield
   192  	 * @param target (MovieClip)
   193  	 * @param style (Object) ConsoleStyle.CSS reference obj
   194  	 * @param str (String) contents of textfield [i.e. "trace"]
   195  	 * @return TextField
   196  	 * {@code Usage:
   197  	 * 	var tf:TextField = ConsoleStyle.$.createStyledTextField(_this, ConsoleStyle.$.CSS.item.textfields.origin, "hello world");
   198  	 * }
   199  	 */
   200  	public function createStyledTextField (target:MovieClip, style:Object, str:String):TextField {
   201  		// if no 'str' arg was passed use default style.t text
   202  		var s:String = (str != undefined) ? str : style.t;
   203  		// if style.c is an Object (instead of number) assume we are a message type and resolve the color based on 'str' arg.
   204  		var c:Number = (!isNaN(style.c)) ? style.c : resolveTypeColor(str);
   205  		// if 'messageTf' || 'originTf' then make autoSize, multiLine, wordWrap, selectable true; else default to false.
   206  		var isSizable:Boolean = (style.n == CSS.item.textfields.message.n || style.n == CSS.item.textfields.origin.n);
   207  		// see if we want to be an html textfield
   208  		var isHtml:Boolean = (style.html == true) ? true : false;
   209  		// see if we want a selectable textfield
   210  		var isSelectable:Boolean = (style.selectable == true) ? true : false;
   211  		// build properties and format objects
   212  		var props:Object = {type:"dynamic", html:isHtml, autoSize:isSizable, wordWrap:isSizable, multiline:isSizable, selectable:isSelectable, mouseWheelEnabled:false, embedFonts:false, _alpha:style.a, htmlText:s};		
   213  		var format:Object = {font:"Arial",size:9, color:c};
   214  		// create the TextField now that we have good properties, formats and string.
   215  		var t:TextField = TextUtils.create(target, style.n, style.x, style.y, style.w, style.h, props, format);
   216  		return t;
   217  	}

   219  	/**
   220  	 * Compare the type string argument to those existing in the ConsoleStyle.COLOR.item.textfields.type style,
   221  	 * if one matches return it, else default to the 'custom' style.
   222  	 * @param type (String) type as string
   223  	 * @return Number
   224  	 * {@code Usage:
   225  	 * 	var c:Number = resolveTypeColor("trace"); // returns CSS.item.textfields.type.c.trace
   226  	 * 	var c:Number = resolveTypeColor("sekati"); // returns CSS.item.textfields.type.c.custom
   227  	 * }
   228  	 */
   229  	private function resolveTypeColor (type:String):Number {
   230  		var t:String = type.toLowerCase();
   231  		var c:Number = (!CSS.item.textfields.type.c[t]) ? CSS.item.textfields.type.c.custom : CSS.item.textfields.type.c[t];
   232  		//trace("resolveTypeColor: "+t+": "+ c);
   233  		return c;
   234  	}
