     9   /**
    10    * recursively trace an objects contents
    11    * 
    12    * {@code Usage:
    13    * var a = ["a", "b", ["aa"], "BB", ["aaa", "BBB"], {joe:[833, 38]}];
    14    * TraceObj.recurse(a);
    15    * }
    16    */
    17  class com.sekati.log.TraceObj {
    18  	/**
    19  	 * recursively trace object
    20  	 * @param obj (Object) target object
    21  	 * @param path (String)
    22  	 * @param level (Number)
    23  	 * @param maxPathLength (Number)
    24  	 * @return Void
    25  	 */
    26  	public static function recurse (obj:Object, path:String, level:Number, maxPathLength:Number):Void {
    27  		var padding:String;
    28  		var paddingChar:String = " ";
    29  		var parentType:String;
    30  		var currentType:String;
    31  		var newPath:String;
    32  		//defaults
    33  		if (level == null) {
    34  			level = 0;
    35  		}
    36  		if (path == null) {
    37  			path = "obj";
    38  		}
    39  		//maxPathLength (only defined initially)   
    40  		if (maxPathLength == null) {
    41  			maxPathLength = paddingRecursion (obj, path) + 3;
    42  		}
    43  		//calculate parents type   
    44  		parentType = (obj instanceof Array) ? "array" : typeof (obj);
    45  		for (var i in obj) {
    46  			//calculate path
    47  			newPath = (parentType == "array") ? path + "[" + i + "]" : path + "." + i;
    48  			//calculate this type
    49  			currentType = (obj[i] instanceof Array) ? "array" : typeof (obj[i]);
    50  			//find how many padding I'll need for this item to print
    51  			padding = "";
    52  			for (var j = 0; j < maxPathLength - newPath.length; j++) {
    53  				padding += paddingChar;
    54  			}
    55  			trace (newPath + padding + obj[i] + "  (" + currentType + ")");
    56  			//go deeper
    57  			recurse (obj[i], newPath, level + 1, maxPathLength);
    58  		}
    59  	}
    60  	//recurse through everything to find what the biggest path string will be - strictly for formatting purposes
    61  	private static function paddingRecursion (obj:Object, path:String, longestPath:Number):Number {
    62  		var parentType:String;
    63  		if (longestPath == null) {
    64  			longestPath = 0;
    65  		}
    66  		//calculate parents type   
    67  		parentType = (obj instanceof Array) ? "array" : typeof (obj);
    68  		for (var i in obj) {
    69  			//this levels path
    70  			var newPath:String = (parentType == "array") ? path + "[" + i + "]" : path + "." + i;
    71  			if (newPath.length > longestPath) {
    72  				longestPath = newPath.length;
    73  			}
    74  			//outside recursion   
    75  			var outsideRecursion:Number = paddingRecursion (obj[i], newPath, longestPath);
    76  			if (outsideRecursion > longestPath) {
    77  				longestPath = outsideRecursion;
    78  			}
    79  		}
    80  		return longestPath;
    81  	}
