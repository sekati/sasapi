    21  /**
    22   * UI Console for attaching or connecting too
    23   */
    24  class com.sekati.log.Console {
    25  	
    26  	// core console props
    27  	private static var _instance:Console;
    28  	private var _this:Console;
    29  	private var _cs:ConsoleStyle;
    30  	private var _style:Object;
    31  	
    32  	// ui props
    33  	private var _console:MovieClip;
    34  	private var _head:MovieClip;
    35  	private var _headTf:TextField;
    36  	private var _fps:MovieClip;
    37  	private var _holder:MovieClip;
    38  	private var _list:MovieClip;
    39  	private var _mask:MovieClip;
    40  	private var _scroll:Scroll;
    41  	private var _gutter:MovieClip;
    42  	private var _bar:MovieClip;
    43  	
    44  	// manager props
    45  	private var _logItems:Array;
    46  	private var _logIndex:Number;
    47  	private var _key:Object;
    48  	private var _rx_lc:LocalConnection;
    49  	
    50  	/**
    51  	 * Singleton Private Constructor
    52  	 */
    53  	private function Console() {
    54  		_this = this;
    55  		_cs = ConsoleStyle.getInstance();
    56  		_style = _cs.CSS;
    57  		Dispatcher.$.addEventListener(LogEvent.onLogEVENT, Delegate.create (_this, onLogEvent));
    58  		logConsoleConnect();
    59  		createUI();
    60  	}
    61  
    62  	/**
    63  	 * Singleton Accessor
    64  	 * @return Console
    65  	 */	
    66  	public static function getInstance():Console {
    67  		if (!_instance) {
    68  			_instance = new Console();
    69  		}
    70  		return _instance;
    71  	}
    72  
    73  	/**
    74  	 * shorthand singleton accessor getter
    75  	 */
    76  	 public static function get $():Console {
    77  	 	return Console.getInstance();	
    78  	 }
    79  	
    80  	/**
    81  	 * Create the core Console UI
    82  	 */
    83  	private function createUI():Void {
    84  		// rect	- createStyledRect (target:MovieClip, style:Object)
    85  		_console = _cs.createStyledRectangle(_level0, _style.console);
    86  		_console._quality = "LOW";
    87  		//_console._visible = false;
    88  		
    89  		_head = _cs.createStyledRectangle(_console, _style.console.head);
    90  		
    91  		// text - createStyledTextField (target:MovieClip, style:Object, str:String)
    92  		_headTf = _cs.createStyledTextField(_head, _style.console.head.textfields.head);
    93  		
    94  		// create ConsoleFPSMonitor
    95  		_fps = ClassUtils.createEmptyMovieClip (com.sekati.log.ConsoleFPSMonitor, _head, _style.console.head.fps.n, {_x:_style.console.head.fps.x, _y:_style.console.head.fps.y});		
    96  		
    97  		// create core ui clip holders
    98  		_holder = _cs.createPositionClip(_console, _style.console.holder);
    99  		_list = _cs.createPositionClip(_holder, _style.console.holder.list);
   100  		
   101  		// create core ui clip shapes
   102  		_mask = _cs.createStyledRectangle(_holder, _style.console.holder.mask);
   103  		_gutter = _cs.createStyledRectangle(_holder, _style.console.holder.gutter);
   104  		_gutter.cacheAsBitmap = true;
   105  		_bar = _cs.createStyledRectangle(_holder, _style.console.holder.bar);
   106  		_bar.cacheAsBitmap = true;
   107  		_bar._visible = false;
   108  		
   109  		// mask item list
   110  		_list.setMask(_mask);
   111  		
   112  		// create scrollable functionality
   113  		_scroll = new Scroll("_y", _list, _mask, _gutter, _bar, true, true, true, true, _list, 0.05, 0.5);
   114  	
   115  		// EVENTS start
   116  		
   117  		// make console draggable
   118  		_head.onPress  = Delegate.create (_console, startDrag);
   119  		_head.onRelease = _head.onReleaseOutside = Delegate.create (_console, stopDrag);
   120  		
   121  		// EVENTS stop
   122  		
   123  		// position the console on stage.
   124  		_console._x = _style.console.startx;
   125  		_console._y = _style.console.starty;
   126  		
   127  		// create the meta item
   128  		ClassUtils.createEmptyMovieClip (com.sekati.log.ConsoleItem, _list, "metaItem", {_x:0, _y:0, _data:{_isMeta:true}});
   129  		
   130  		//runItemTest();
   131  		
   132  		// keylistener
   133  		_key = new Object ();
   134  		_key.onKeyDown = Delegate.create(_instance, key_onKeyDown);
   135  		/*
   136  		_key.onKeyDown = function ():Void  {
   137  			if (Key.isDown (Key.UP) && Key.isDown (Key.LEFT)) {
   138  				trace("key");
   139  				_console._visible = (!_console._visible) ? true : false;
   140  			}
   141  		};
   142  		*/
   143  		Key.addListener (_key);		
   144  	}
   145  	
   146  	/**
   147  	 * Key Handler for all Console hotkeys.
   148  	 * @return Void
   149  	 */
   150  	private function key_onKeyDown(code:Number):Void {
   151  		//trace("keydel: "+code);
   152  		if (Key.isDown (Key.UP) && Key.isDown (Key.LEFT)) {
   153  			trace("key");
   154  			_console._visible = (!_console._visible) ? true : false;
   155  		}		
   156  	}
   157  	
   158  	/**
   159  	 * Create a LocalConnection to Logger
   160  	 * @return Void
   161  	 */
   162  	private function logConsoleConnect():Void {
   163  		_rx_lc = new LocalConnection();
   164  		_rx_lc[LogConsoleConnector.methodName]= Delegate.create(_this, addItem);
   165  		_rx_lc.connect(LogConsoleConnector.connectionName);
   166  	}
   167  	
   168  	/**
   169  	 * Add and return a new ConsoleItem
   170  	 * @param data (Object) ConsoleItem._data = {id:Number, type:String, origin:String, message:String, benchmark:Number, _isMeta:Boolean}
   171  	 * @return 	MovieClip
   172  	 */
   173  	public function addItem (data:Object):MovieClip {
   174  		var item:MovieClip = ClassUtils.createEmptyMovieClip (com.sekati.log.ConsoleItem, _list, "consoleItem_"+data.id, {_x:0, _y:_list._height, _data:data});
   175  		updateScroll(item);
   176  		return item;
   177  	}
   178  	
   179  	/**
   180  	 * Update the {@link Scroll} to the last added item if we are not mousing the area.
   181  	 * @param item (MovieClip) last added item
   182  	 * @return Void
   183  	 */
   184  	public function updateScroll(item:MovieClip):Void {
   185  		if(!_scroll.isMouseInArea() && !_scroll.isDrag && _console._visible == true) {
   186  			_global["setTimeout"](_scroll,"slideContent",50, item._y+item._height, 0.1);
   187  		}
   188  	}
   189  	
   190  	/**
   191  	 * Handle events from Logger and parse them to addItem
   192  	 * @param eventObj (Event)
   193  	 * @return Void
   194  	 */ 
   195  	private function onLogEvent (eventObj:Event):Void {
   196  		//trace ("eventObj{target:" + eventObj.target + ",type:" + eventObj.type + ",message:" + eventObj.data.message + "};");
   197  		addItem(eventObj.data);
   198  	}
   199  	
   200  	/**
   201  	 * Clean and destroy
   202  	 * @return Void
   203  	 */
   204  	public function destroy():Void {
   205  		Dispatcher.$.removeEventListener("LOG_EVENT",Delegate.create (_this, onLogEvent));
   206  		_rx_lc.close();
   207  		_fps.destroy();
   208  	}
   209  }
