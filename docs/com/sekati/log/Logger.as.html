    15  /**
    16   * Logger is a multi-tiered debugging tool designed to clarify the debugging process.
    17   * 
    18   * The Logger consists of two main tiered components:
    19   * 	- "levels" (method calls, i.e. 'trace', 'info', 'status', etc).
    20   * 	- "filters" (origin of the level call, i.e. '_root', 'myClassInstance', 'myMc', etc).
    21   * 	
    22   * New level (methods) may be created dynamically "overloading" the singleton class 
    23   * via {@link setLevel}, __resolve & proxy as illustrated in the example usage below.
    24   * 
    25   * Logger can be configured to display its output in several different ways:
    26   * 	- Flash IDE Output Panel.
    27   * 	- {@link Console} hidden inside compiled swf.
    28   * 	- {@link Console} via {@link http://debug.sekati.com} localConnection.
    29   * 
    30   * @see {@link com.sekati.log.Console}
    31   */
    32  class com.sekati.log.Logger implements CoreInterface {
    33  
    34  	private static var _instance:Logger;
    35  
    36  	private var _levels:Object;
    37  	private var _filters:Array;
    38  	private var _proxyObj;
    39  	private var _watch:StopWatch;	
    40  	private var _isEnabled:Boolean;
    41  	private var _logId:Number;
    42  	
    43  	// default level stubs
    44  	public var trace:Function;
    45  	public var info:Function;
    46  	public var status:Function;
    47  	public var warn:Function;
    48  	public var error:Function;
    49  	public var fatal:Function;
    50  
    51  	/**
    52  	 * Singleton Private Constructor
    53  	 */
    54  	private function Logger() {
    55  		resetLevels();
    56  		resetFilters();
    57  		_proxyObj = new Object();
    58  		_watch = new StopWatch(true);
    59  		_isEnabled = true;
    60  		_logId = 0;
    61  	}
    62  
    63  	/**
    64  	 * Singleton Accessor
    65  	 * @return Logger
    66  	 */	
    67  	public static function getInstance ():Logger {
    68  		if (!_instance) {
    69  			_instance = new Logger();
    70  		}
    71  		return _instance;
    72  	}

    75  	 * Shorthand singleton accessor getter

    77  	 */

    80  	 }

    83  	// Core Controllers
    84  	
    85  	/**
    86  	 * enabled setter
    87  	 * @param b (Boolean) enable or disable Logger
    88  	 */
    89  	public function set enabled (b:Boolean):Void {
    90  		_isEnabled = b;
    91  	}
    92  
    93  	/**
    94  	 * enabled getter
    95  	 * @return Boolean
    96  	 */	
    97  	public function get enabled ():Boolean {
    98  		return _isEnabled;
    99  	}
   100  
   101  	/**
   102  	 * level and filter _status getter
   103  	 * @return String
   104  	 */
   105  	public function get _status ():String {
   106  		return getLevels () + "\n" + getFilters ();
   107  	}
   108  
   109  	/**
   110  	 * reset Out to default levels and filters
   111  	 * @return Void
   112  	 */
   113  	public function reset ():Void {
   114  		setAllLevels (true);
   115  		resetFilters ();
   116  	}
   117  	
   118  	//////////////////////////////////////////////////////////////
   119  	// Level Handlers
   120  
   121  	/**
   122  	 * Enable/disable a level and create the level if it does not already exist.
   123  	 * @param level (String) level name
   124  	 * @param isEnabled (Boolean) enabled status
   125  	 * @return Void
   126  	 */
   127  	public function setLevel (level:String, isEnabled:Boolean):Void {
   128  		_levels[level.toLowerCase()] = isEnabled;
   129  	}
   130  
   131  	/**
   132  	 * Enable or disable all existing levels.
   133  	 * @param isEnabled (Boolean) enabled status
   134  	 * @return Void
   135  	 */
   136  	public function setAllLevels (isEnabled:Boolean):Void {
   137  		for (var i in _levels) {
   138  			setLevel (_levels[i],isEnabled);
   139  		}
   140  	}
   141  
   142  	/**
   143  	 * Reset all levels (clearing previously created levels)
   144  	 * @return Void
   145  	 */
   146  	public function resetLevels():Void {
   147  		_levels = {trace:true, info:true, status:true, warn:true, error:true, fatal:true, object:true};
   148  	}
   149  
   150  	/**
   151  	 * Returns a stringified overview of all levels statuses.
   152  	 * @return String
   153  	 */
   154  	public function getLevels ():String {
   155  		var a:Array = new Array ();
   156  		for (var i in _levels) {
   157  			a.push (i + ":" + _levels[i].toString ());
   158  		}
   159  		return "_levels={" + a.toString () + "};";
   160  	}
   161  	
   162  	//////////////////////////////////////////////////////////////
   163  	// Filter Handlers
   164  	
   165  	/**
   166  	 * Enable/disable a filter and create the filter if it does not already exist.
   167  	 * @param origin (Object) object to filter on [usually a string]
   168  	 * @param isFiltered (Boolean)
   169  	 * @return Void
   170  	 */
   171  	public function setFilter (origin:Object, isFiltered:Boolean):Void {
   172  		if (isFiltered) {
   173  			filter(origin);
   174  		} else {
   175  			unfilter(origin);
   176  		}
   177  	}
   178  
   179  	/**
   180  	 * Returns a stringified overview of all filters statuses.
   181  	 * @return String
   182  	 */
   183  	public function getFilters ():String {
   184  		return "_filters=[" + _filters.toString () + "];";
   185  	}
   186  
   187  	/**
   188  	 * Reset all filters (clearing previous filters)
   189  	 * @return Void
   190  	 */
   191  	public function resetFilters ():Void {
   192  		_filters = [];
   193  	}
   194  	
   195  	/**
   196  	 * Check if an origin's output is being filtered
   197  	 * @param origin (Object) to check.
   198  	 * @return Boolean
   199  	 */
   200  	public function isFiltered (origin:Object):Boolean {
   201  		var o:String = String (origin);
   202  		for (var i:Number = 0; i < _filters.length; i++) {
   203  			if (_filters[i] == o) {
   204  				return true;
   205  			}
   206  		}
   207  		return false;
   208  	}	
   209  	
   210  	/**
   211  	 * Add an origin to the filters array.
   212  	 * @param origin (Object) to be added.
   213  	 * @return Void
   214  	 */
   215  	private function filter (origin:Object):Void {
   216  		var o:String = String (origin);
   217  		if (!isFiltered (o)) {
   218  			_filters.push (o);
   219  		}
   220  	}
   221  
   222  	/**
   223  	 * Remove an origin from the filters array.
   224  	 * @param origin (Object) to be removed.
   225  	 * @return Void
   226  	 */
   227  	private function unfilter (origin:Object):Void {
   228  		var o:String = String (origin);
   229  		for (var i:Number = 0; i < _filters.length; i++) {
   230  			if (_filters[i] == o) {
   231  				_filters.splice (i,1);
   232  				break;
   233  			}
   234  		}
   235  	}
   236  
   237  	/**
   238  	 * Currently unused; AS3 has getNameSpace - when Logger is ported origin will change from string/mc to class  namespace.
   239  	 * For now it may be better to keep filtering on object/string rather than class(?)
   240  	 * @param origin (Object)
   241  	 * @return Void
   242  	 */
   243  	private function resolveOrigin (origin):Void {
   244  		var o = (typeof (origin) == "string") ? origin : origin._classname;
   245  		if (!o) {
   246  			o = origin;
   247  		}
   248  		trace (o);
   249  	}

   251  	//////////////////////////////////////////////////////////////
   252  	// Output Handlers
   253  	
   254  	private function _output (level:String, origin, msg):Void {
   255  		if (_isEnabled == false || _levels[level] == false || isFiltered (origin) == true) {
   256  			return;
   257  		}		
   258  		var benchmark:Number = _watch.lap();
   259  		var id:Number = _logId++;
   260  		var e:Event = new Event("LOG_EVENT",this,{id:id, type:level.toLowerCase(), origin:String(origin), message:String(msg), benchmark:benchmark});
   261  		Dispatcher.$.dispatchEvent(e);
   262  		/*
   263  		var o:String = level + ":[" + String (origin) + "]: " + msg;
   264  		 if(!_isPanel) {
   265  			trace(o);
   266  		} else {
   267  			var e:Event = new Event("OUT_EVENT",this,{message:o, level:level});
   268  			Dispatcher.$.dispatchEvent(e);
   269  		}
   270  		 */
   271  	}

   274  	//////////////////////////////////////////////////////////////
   275  	// Level Overloading	
   276  
   277  	// __resolve catches all wrapper & invented levels and processes them thru the proxy to _output: cool!
   278  	private function __resolve (name:String):Function {
   279  		if (name.indexOf ("LOG_EVENT") > 1) {
   280  			return;
   281  		}
   282  		var f:Function = function() {
   283  			arguments.unshift (name);
   284  			return __proxy.apply (_proxyObj, arguments);
   285  		};
   286  		_proxyObj[name] = f;
   287  		return f;
   288  	}
   289  
   290  	private function __proxy (name:String) {
   291  		arguments.shift ();
   292  		var n:String = String (name).toUpperCase ();
   293  		var o:String = String (arguments[0]);
   294  		var s:String = String (arguments[1]);
   295  		_instance._output (n,o,s);
   296  	}
   297  	
   298  	//////////////////////////////////////////////////////////////
   299  	// Special Level Method(s)
   300  	
   301  	/**
   302  	 * object is a special level (method) which handles object recursion via {@link com.sekati.log.Inspector}
   303  	 * @param origin (Object) origin for filtering purposes
   304  	 * @param obj (Object) object to be recursed thru Out
   305  	 * @return Void 
   306  	 */
   307  	public function object (origin:Object, obj:Object):Void {
   308  		var insp:Inspector = new Inspector(obj,origin);
   309  		_output ("OBJECT", origin, insp);
   310  	}
