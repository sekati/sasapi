     9  import mx.services.SOAPCall;
    10  import mx.services.WebService;
    11  
    12  /**
    13   * Soap Client class to be used with JNuSOAP.<br>
    14   * 
    15   * <br>Q: 'There are multiple possible ports in the WSDL file; please specify a service name and port name!?'
    16   * <br>A: add port to class instance, seach for "port" in the wsdl & see url's below for more info.
    17   * 
    18   * @see <a href="http://www.intangibleinc.com/movabletype/archives/000007.html">http://www.intangibleinc.com/movabletype/archives/000007.html</a>
    19   * @see <a href="http://livedocs.macromedia.com/fms/2/docs/wwhelp/wwhimpl/common/html/wwhelp.htm?context=LiveDocs_Parts&file=00000796.html">http://livedocs.macromedia.com/fms/2/docs/wwhelp/wwhimpl/common/html/wwhelp.htm?context=LiveDocs_Parts&file=00000796.html</a>
    20   * @see <a href="http://www.flash-db.com/Tutorials/lclasses/lclasses.php?page=2">http://www.flash-db.com/Tutorials/lclasses/lclasses.php?page=2</a>
    21   * @see <a href="http://www.adobe.com/devnet/flash/articles/flmxpro_webservices_03.html">http://www.adobe.com/devnet/flash/articles/flmxpro_webservices_03.html</a>
    22   */
    23  class com.sekati.service.SoapClient {
    24  
    25  	private var _ws:WebService;
    26  	private var _wsdl:String;
    27  	private var debugFn:Function;
    28  
    29  	/**
    30  	 * Constructor
    31  	 * @param wsdl (String) wsdl url
    32  	 * @param port (String) optional service port
    33  	 * @param debugHandler (Function) debug function
    34  	 * @param loadFn (Function) service onLoad handler
    35  	 * @param faultFn (Function) service onFault handler
    36  	 * @return Void
    37  	 */
    38  	public function SoapClient(wsdl:String, port:String, debugHandler:Function, loadFn:Function, faultFn:Function) {
    39  		// set passed wsdl to class
    40  		_wsdl = wsdl;
    41  		// set passed debug function to class
    42  		debugFn = debugHandler;
    43  		// throw instatiation notice
    44  		debugFn("SoapClient Initialized ...\n");
    45  		// attempt webservice connection
    46  		wsConnect(loadFn, faultFn, port);
    47  	}
    48  	
    49  	private function wsConnect(loadFn:Function, faultFn:Function, port:String):Void {
    50  		// instantiate new webservice
    51  		_ws = new WebService(_wsdl);
    52  		// added re http://www.adobe.com/devnet/flash/articles/flmxpro_webservices_03.html
    53  		if (port) {
    54  			_ws._portName = port;
    55  		}
    56  		// alias debug function to prevent scope issue else onLoad/onFault events call debugFn as ws method 
    57  		_ws.debugFn = debugFn;
    58  		// webservice load handler
    59  		_ws.onLoad = function(wsdl:String):Void  {
    60  			debugFn("@@@@@ WEBSERVICE LOADING / PARSING WSDL @@@@@\n"+wsdl+"\n");
    61  			loadFn();
    62  		};
    63  		// webservice fault handler
    64  		_ws.onFault = function(fault:Object):Void  {
    65  			debugFn("##### WEBSERVICE FAULTING #####");
    66  			debugFn("fault.faultstring: "+fault.faultstring);
    67  			debugFn("fault.faultcode: "+fault.faultcode);
    68  			debugFn("fault.detail: "+fault.detail);
    69  			debugFn("fault.faultactor: "+fault.faultactor);
    70  			faultFn();
    71  		};
    72  	}
    73  	
    74  	/**
    75  	 * webservice method call
    76  	 * @param method (String) service method name
    77  	 * @param args (Array) array of service method arguments
    78  	 * @param resultFn (Function) onResult handler
    79  	 * @param faultFn (Function) onFault handler
    80  	 * @return Void
    81  	 */
    82  	public function wsCall(method:String, args:Array, resultFn:Function, faultFn:Function):Void {
    83  		// Call Remote WebService Method using args array; then call defined resultFn or FaultFn 
    84  		// NOTE: webservice method invocations return an asynchronous callback. 
    85  		// callback is undefined if the service itself is not created (and service.onFault fires)
    86  		var cb:Function = _ws[method].apply(_ws, args);
    87  		// alias debug function for handler to prevent scope issue else onResult/onFault events call debugFn as ws method
    88  		cb.debugFn = debugFn;
    89  		// call load handler
    90  		cb.onResult = function(result):Void  {
    91  			// receive SOAP output, which in is deserialized as a struct (ActionScript object)
    92  			debugFn("@@@@@ CALL RESULT @@@@@");
    93  			debugFn(result);
    94  			// debug request & response xml objects
    95  			debugFn("SOAP REQUEST ENVELOPE: "+cb.request+"\n");
    96  			debugFn("SOAP RESPONSE ENVELOPE: "+cb.response+"\n");
    97  			// debug myCall properties and info
    98  			for (var i in this.myCall) {
    99  				debugFn("MyCall SoapCall Properties: "+i+"  "+this.myCall[i]);
   100  			}
   101  			for (var i in SOAPCall) {
   102  				debugFn("SOAPCall-> "+i+" "+SOAPCall[i]);
   103  			}
   104  			//SOAPCall.doDecoding
   105  			debugFn("\n");
   106  			resultFn(result);
   107  		};
   108  		// call fault handler
   109  		cb.onFault = function(fault):Void  {
   110  			// catch SOAP fault and handle it according to this app's requirements
   111  			debugFn("##### CALL FAULT #####");
   112  			debugFn("fault.faultstring: "+fault.faultstring);
   113  			debugFn("fault.faultcode: "+fault.faultcode);
   114  			debugFn("fault.detail: "+fault.detail);
   115  			debugFn("fault.faultactor: "+fault.faultactor);
   116  			debugFn("REQUEST: "+cb.request);
   117  			debugFn("SOAP RESPONSE ENVELOPE: "+cb.response);
   118  			faultFn(fault);
   119  		};
   120  	}
   121  }
