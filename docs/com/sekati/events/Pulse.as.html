     9  import com.sekati.core.FWDepth;
    10  import com.sekati.utils.ClassUtils;
    11  import com.sekati.utils.Delegate;
    12   
    13  /**
    14   * Framerate Pulse Broadcaster
    15   */
    16  class com.sekati.events.Pulse {
    17  	
    18  	// AsBroadcaster stubs
    19  	private var addListener:Function;
    20  	private var removeListener:Function;
    21  	private var broadcastMessage:Function;
    22  	private var _listeners:Array;
    23  		
    24  	private static var _instance:Pulse;
    25  	private var _mc:MovieClip;
    26  	private var _f:Function;
    27  	
    28  	public static var onEnterFrameEVENT:String = "_onEnterFrame";
    29  	
    30  	/**
    31  	 * Singleton Private Constructor: initializes centralized management of mx.events.EventDispatcher
    32  	 */
    33  	private function Pulse() {
    34  		AsBroadcaster.initialize(this);
    35  		_mc = ClassUtils.createEmptyMovieClip(com.sekati.display.BaseClip, _level0, "___FramePulse", {_depth:FWDepth.Pulse});
    36  		_f = Delegate.create(this, broadcastMessage, onEnterFrameEVENT);
    37  	}
    38  
    39  	/**
    40   	 * Singleton Accessor
    41  	 * @return Dispatcher
    42  	 */
    43  	public static function getInstance():Pulse {
    44  		if (!_instance) {
    45  			_instance = new Pulse();
    46  		}
    47  		return _instance;
    48  	}
    49  	
    50  	/**
    51  	 * Shorthand singleton accessor getter
    52  	 */
    53  	 public static function get $():Pulse {
    54  	 	return Pulse.getInstance();	
    55  	 }
    56  	
    57  	/**
    58  	 * Start OEF Pulse
    59  	 * @return Void
    60  	 */
    61  	public function start():Void {
    62  		_mc.onEnterFrame = _f;		
    63  	}	
    64  	
    65  	/**
    66  	 * Stop OEF Pulse
    67  	 */
    68  	public function stop():Void {
    69  		_mc.onEnterFrame = null;
    70  	}
    71  	
    72  	/**
    73  	 * Check if OEF Pulse is running
    74  	 * @return Boolean
    75  	 */
    76  	public function isRunning():Boolean{
    77  		return _mc.onEnterFrame == _f;
    78  	}	
    79  
    80  	/**
    81  	 * Destroy OEF Pulse
    82  	 * @return Void
    83  	 */
    84  	public function destroy():Void {
    85  		_instance.stop();
    86  		_mc.destroy();
    87  		delete _instance;
    88  	}
    89  	
    90  	/**
    91  	 * Add a Pulse listener and auto-start Pulse if not running
    92  	 * @param o (Object) to register
    93  	 * @return Void
    94  	 */
    95  	public function addFrameListener(o:Object):Void{
    96  		if (_listeners.length < 1) start();
    97  		addListener(o);
    98  	}
    99  	
   100  	/**
   101  	 * Add an array of Pulse listeners
   102  	 * @param a (Array) of objects to register
   103  	 * @return Void
   104  	 */
   105  	public function addFrameListeners(a:Array):Void {
   106  		for(var i:Number=0; i<a.length; i++) {
   107  			addFrameListener(a[i]);	
   108  		}	
   109  	}
   110  	
   111  	/**
   112  	 * remove a Pulse listener and auto-stop Pulse if no listeners remain
   113  	 * @param o (Object) to unregister
   114  	 * @return Void
   115  	 */
   116  	public function removeFrameListener(o:Object):Void{
   117  		removeListener(o);
   118  		if (_listeners.length < 1) stop();
   119  	}
   120  
   121  	/**
   122  	 * Remove an array of Pulse listeners
   123  	 * @param a (Array) of objects to unregister
   124  	 * @return Void
   125  	 */
   126  	public function removeFrameListeners(a:Array):Void {
   127  		for(var i:Number=0; i<a.length; i++) {
   128  			removeFrameListener(a[i]);	
   129  		}	
   130  	}
   131  
   132  	/**
   133  	 * check is a Pulse listener is registered
   134  	 * @return Boolean
   135  	 */
   136  	public function isListenerRegistered(o:Object):Boolean{
   137  		for (var i in _listeners) if(_listeners[i] === o) return true;
   138  		return false;
   139  	}
   140  }
