    13  /**
    14   * Sort an array of Objects positions into various shapes and patterns.

    16  class com.sekati.geom.Sort {

    18  	private var _items:Array;
    19  	private var _sort:Array;
    20  	
    21  	/**
    22  	 * Sort Constructor
    23  	 * @param items (Array)
    24  	 * @throws Error on invalid items argument array
    25  	 * {@code Usage:
    26  	 * 	var sort:Sort = new Sort([mc0, mc1, mc2, mc3]);
    27  	 * }
    28  	 */
    29  	public function Sort(items:Array) {
    30  		// verify that each item in the array has an _x,_y or x,y property to be sorted.
    31  		for (var i:Number = 0; i < items.length; i++) {
    32  			if((!items[i]._x && !items[i]._y) && (!items[i].x && !items[i].y)) {
    33  				throw new Error("@@@ com.sekati.geom.Sort Error: constructor expects 'items' argument array to contain objects with '_x',_y' or 'x','y' properties.");
    34  				return;
    35  			}
    36  		}
    37  		_items = items;
    38  		_sort = new Array();
    39  	}
    40  
    41  	/**
    42  	 * Sort items in grid.
    43  	 * @param start (Point)
    44  	 * @param numPerRow (Number)
    45  	 * @param offset (Number)
    46  	 * @return Array
    47  	 * {@code Usage:
    48  	 * 	var positions:Array = new Sort(itemArr).grid(new Point(100,100), 10, 1);
    49  	 * }
    50  	 */	
    51  	public function grid (start:Point, numPerRow:Number, offset:Number):Array {
    52  		var _xpos:Number = 0;
    53  		var _ypos:Number = 0;
    54  		var _offset:Number = (!offset) ? 0 : offset;
    55  		
    56  		// if there is no _width,_height lets assume the items are Points and count them as one pixel.
    57  		var _w:Number = (!_items[0]._width) ? 1 : _items[0]._width;
    58  		var _h:Number = (!_items[0]._height) ? 1 : _items[0]._height;
    59  		var _numPerRow:Number = (!numPerRow) ? Math.ceil((Stage.width-start.x)/(_w+_offset)) : numPerRow;
    60  		
    61  	    for (var i:Number=0; i<_items.length; i++) {
    62  	        if ((i%numPerRow) == 0) {
    63  	            _xpos = _offset;
    64  	            _ypos += _h + _offset;
    65  	        } else {
    66  	            _xpos += _w + _offset;
    67  	            _ypos += 0;
    68  	        }
    69  	        _sort[i] = new Point(Math.round(_xpos)+start.x, Math.round(_ypos)+start.y);
    70  	    }
    71  	    return _sort;		
    72  	}

    74  	/**
    75  	 * Sort items in circle.
    76  	 * @param center (Point)
    77  	 * @param radius (Number)
    78  	 * @return Array
    79  	 * {@code Usage:
    80  	 * 	var positions:Array = new Sort(itemArr).circle(new Point(250,250), 255);
    81  	 * }
    82  	 */
    83  	public function circle(center:Point, radius:Number):Array {
    84  		for (var i:Number = 0; i< _items.length; i++) {
    85  			var angle:Number = i*(360/_items.length);
    86  			var x:Number = Math.round((center.x+(radius*Math.cos((angle-180)*Math.PI/180))));
    87  			var y:Number = Math.round((center.y+(radius*Math.sin((angle-180)*Math.PI/180))));
    88  			_sort[i] = new Point(x,y);
    89  		}
    90  		return _sort;
    91  	}

    93  	/**
    94  	 * Sort items in sine wave.
    95  	 * @param waves (Number)
    96  	 * @param width (Number) sine wave width
    97  	 * @param width (Number) sine wave y position 
    98  	 * @param widthCap (Number) use max 90% of the width
    99  	 * @param heightCap (Number) use max 60% of the height
   100  	 * @return Array
   101  	 * {@code Usage:
   102  	 * 	var positions:Array = new Sort(itemArr).sine(1.5, Stage.width, Stage.height/2, 0.95, 0.5);
   103  	 * }
   104  	 */
   105  	public function sine(waves:Number, width:Number, yPos:Number, widthCap:Number, heightCap:Number):Array {
   106  	    var _degStep:Number = (waves * 2*Math.PI)/(_items.length-1);
   107  	    var _xStep:Number = (width*widthCap)/_items.length;
   108  	    for (var i:Number = 0; i < _items.length; i++){
   109  	        var x:Number = Math.round((width-width*widthCap)/2+_xStep*i);
   110  	        var y:Number = Math.round(Math.sin(_degStep*i)*yPos*heightCap+yPos);
   111  	        _sort[i] = new Point(x,y);
   112  	    }
   113  	    return _sort;
   114  	}

   116  	/**
   117  	 * Sort items in triangle.
   118  	 * @param center (Point)
   119  	 * @param sideLength (Number)
   120  	 * @return Array
   121  	 * {@code Usage:
   122  	 * 	var positions:Array = new Sort(itemArr).triangle(new Point(250,250), 500);
   123  	 * }
   124  	 */
   125  	public function triangle(center:Point, sideLength:Number):Array {
   126  		var cY:Number = center.y-(sideLength/2);
   127  	    var perimeter:Number = sideLength*3;
   128  	    var objBySide:Number = Math.floor(_items.length/3);
   129  	    var leftObj:Number = _items.length-objBySide*3;
   130  	    var posArray:Array = new Array();
   131  	    var radAngle:Number = Math.PI/3;
   132  	    
   133  	    for (var i:Number=0; i<_items.length; i++) {
   134  	        posArray[i] = new Point();
   135  	        if (i<objBySide){
   136  	            var spacing:Number = sideLength/objBySide;
   137  	            posArray[i].x = (spacing*i)*Math.cos(radAngle);
   138  	            posArray[i].y = (spacing*i)*Math.sin(radAngle);
   139  	        }
   140  	        if (i>=objBySide && i<objBySide*2+Math.ceil(leftObj/2)) {
   141  	            var spacing:Number = (leftObj != 0) ? sideLength/(objBySide+1) : sideLength/objBySide;
   142  	            posArray[i].x = (sideLength*Math.cos(radAngle))-(spacing*(i-objBySide));
   143  	            posArray[i].y = sideLength*Math.sin(radAngle);
   144  	        }
   145  	        if (i>=objBySide*2+Math.ceil(leftObj/2)) {
   146  	            var spacing:Number = (leftObj == 2) ? sideLength/(objBySide+1) : sideLength/objBySide;
   147  	            posArray[i].x = -sideLength*Math.cos(radAngle)+(spacing*(i-(objBySide*2+Math.ceil(leftObj/2))))*Math.cos(radAngle);
   148  	            posArray[i].y = sideLength*Math.sin(radAngle)-(spacing*(i-(objBySide*2+Math.ceil(leftObj/2))))*Math.sin(radAngle);
   149  	        }
   150  			var x:Number = Math.round(center.x+posArray[i].x);
   151  			var y:Number = Math.round(cY+posArray[i].y);
   152  			_sort[i] = new Point(x,y);
   153  	    }
   154  		return _sort;
   155  	}

   157  	/**
   158  	 * Sort items in flower.
   159  	 * @param center (Point)
   160  	 * @param radius (Number)
   161  	 * @return Array
   162  	 * {@code Usage:
   163  	 * 	var positions:Array = new Sort(itemArr).flower(new Point(250,250), 250);
   164  	 * }
   165  	 */	public function flower (center:Point, radius:Number):Array {
   166  		var posArray:Array = new Array();
   167  		var step:Number = (2*Math.PI)/_items.length;
   168  		var b:Number = 1;
   169  		var k:Number = 2;
   170  		for (var i:Number=0; i<_items.length; i++) {
   171  			var r:Number = b*Math.cos(k*i*step);
   172  			posArray[i] = new Point();
   173  			posArray[i].x = (r*Math.cos(i*step))*radius;
   174  			posArray[i].y = (r*Math.sin(i*step))*radius;
   175  			var x:Number = Math.round(center.x+posArray[i].x);
   176  			var y:Number = Math.round(center.y+posArray[i].y);
   177  			_sort[i] = new Point(x,y);
   178  		}
   179  		return _sort;
   180  	}

   182  	/**
   183  	 * Sort items in hedron (star,square,triangle,hexagon,octagon,etc)
   184  	 * @param center (Point)
   185  	 * @param heightCap (Number) use max 90% of the height [default: 0.8]
   186  	 * @param corners (Number)
   187  	 * @param rotate  (Number)
   188  	 * @param isIntraRadial (Boolean) if true star shapes will be drawn else closed hedrons [default: false]
   189  	 * @return Array
   190  	 * {@code Usage:
   191  	 * 	var positions:Array = new Sort(itemArr).hedron(new Point(250,250), 0.6, 8, 90); // create a octagon and rotate 90 degrees.
   192  	 * }
   193  	 */
   194  	public function hedron (center:Point, heightCap:Number, corners:Number, rotate:Number, isIntraRadial:Boolean):Array {
   195  	    var heightCap:Number = (!heightCap) ? 0.8 : heightCap;
   196  	    var outerRadius:Number = center.y * heightCap;
   197  	    var innerRadius:Number = (!isIntraRadial) ? outerRadius * Math.sqrt((1+Math.cos((2*Math.PI)/corners))/2) : outerRadius/2;
   198  	    var cursors:Number = _items.length;
   199  	    var lines:Number = corners * 2;
   200  	    var cursorsPerLine:Number = Math.floor(cursors / lines);
   201  	    var cursorsLeftOver:Number = cursors - cursorsPerLine * lines; // cursors for adoption...
   202  	    var alphaStep:Number = (2*Math.PI) / corners;
   203  	    rotate = (Math.PI/180) * rotate;
   204  	    var alpha:Number = rotate;
   205  	    var actualCursor:Number = 0;
   206  	    for (var i:Number = 0; i < corners; i++) {
   207  	        var xOuter1:Number = Math.sin(alpha) * outerRadius;
   208  	        var yOuter1:Number = Math.cos(alpha) * outerRadius;
   209  	        var xInner:Number = Math.sin(alpha - alphaStep/2) * innerRadius;
   210  	        var yInner:Number = Math.cos(alpha - alphaStep/2) * innerRadius;
   211  	        var xOuter2:Number = Math.sin(alpha - alphaStep) * outerRadius;
   212  	        var yOuter2:Number = Math.cos(alpha - alphaStep) * outerRadius;
   213  	        // plot first line
   214  	        var cursorsOnActualLine:Number = cursorsPerLine;
   215  	        if (cursorsLeftOver > 0) {
   216  	            cursorsOnActualLine++; // add one of the left over cursors on the line
   217  	            cursorsLeftOver--; // one was adopted!
   218  	        }
   219  	        var xStep:Number = (xInner - xOuter1) / cursorsOnActualLine;
   220  	        var yStep:Number = (yInner - yOuter1) / cursorsOnActualLine;
   221  	        var xAct:Number = center.x + xOuter1;
   222  	        var yAct:Number = center.y + yOuter1;
   223  	        for (var j:Number = 0; j < cursorsOnActualLine; j++) {
   224  	            _sort[actualCursor] = new Point(int(xAct), int(yAct));
   225  	            xAct += xStep;
   226  	            yAct += yStep;
   227  	            actualCursor++;
   228  	        }
   229  	        // plot second line
   230  	        var cursorsOnActualLine:Number = cursorsPerLine;
   231  	        if (cursorsLeftOver > 0) {
   232  	            cursorsOnActualLine++; // add one of the left over cursors on the line
   233  	            cursorsLeftOver--; // another one was adopted!
   234  	        }
   235  	        var xStep:Number = (xOuter2 - xInner) / cursorsOnActualLine;
   236  	        var yStep:Number = (yOuter2 - yInner) / cursorsOnActualLine;
   237  	        var xAct:Number = center.x + xInner;
   238  	        var yAct:Number = center.y + yInner;
   239  	        for (var j:Number = 0; j < cursorsOnActualLine; j++) {
   240  	             _sort[actualCursor] = new Point(int(xAct), int(yAct));
   241  	            xAct += xStep;
   242  	            yAct += yStep;
   243  	            actualCursor++;
   244  	        }
   245  	        alpha -= alphaStep;
   246  	    }
   247  	    return _sort;
   248  	}

   291  	/**
   292  	 * Unsort items in random arrangement.
   293  	 * @param topLeft (Point)
   294  	 * @param bottomLeft (Point)
   295  	 * @return Array
   296  	 * {@code Usage:
   297  	 * 	var positions:Array = new Sort(itemArr).unsort(new Point(0,0), new Point(Stage.width, Stage.height));
   298  	 * }

   300  	public function unsort(topLeft:Point, bottomRight:Point):Array {
   301  		for (var i:Number=0; i<_items.length; i++){
   302  			_sort[i] = new Point(MathBase.rnd(topLeft.x,bottomRight.x), MathBase.rnd(topLeft.y,bottomRight.y));
   303  		}
   304  		return _sort;
   305  	}
