    14  	/**
    15  	 * search for key in string
    16  	 * @param str (String)
    17  	 * @param key (String)
    18  	 * @return Boolean
    19  	 */
    20  	public static function search (str:String, key:String):Boolean {
    21  		return (str.indexOf (key) <= -1) ? false : true;
    22  	}

    24  	/**
    25  	 * replace every instance of a string with something else
    26  	 * @param str (String)
    27  	 * @param oldChar (String)
    28  	 * @param newChar (String)
    29  	 * @return String
    30  	 */
    31  	public static function replace (str:String, oldChar:String, newChar:String):String {
    32  		return str.split (oldChar).join (newChar);
    33  	}

    40  	public static function removeSpaces (str:String):String {
    41  		return replace (str, " ", "");
    42  	}

    44  	/**
    45  	 * remove spaces at end and beginning of the string only
    46  	 * @param str (String)
    47  	 * @return String
    48  	 */
    49  	public static function trim (str:String):String {
    50  		var index0:Number = 0;
    51  		while (str.charAt (index0) == " ") {
    52  			index0++;
    53  		}
    54  		var index1:Number = str.length - 1;
    55  		while (str.charAt (index1) == " ") {
    56  			index1--;
    57  		}
    58  		return str.substring (index0, index1 + 1);
    59  	}

    61  	/**
    62  	 * remove spaces  tabs, line feeds, carrige returns from string
    63  	 * @param str (String)
    64  	 * @return String
    65  	 */
    66  	public static function xtrim (str:String):String {
    67  		var o:String = new String ();
    68  		var TAB:Number = 9;
    69  		var LINEFEED:Number = 10;
    70  		var CARRIAGE:Number = 13;
    71  		var SPACE:Number = 32;
    72  		for (var i:Number = 0; i < str.length; i++) {
    73  			if (str.charCodeAt (i) != SPACE && str.charCodeAt (i) != CARRIAGE && str.charCodeAt (i) != LINEFEED && str.charCodeAt (i) != TAB) {
    74  				o += str.charAt (i);
    75  			}
    76  		}
    77  		return o;
    78  	}

    85  	public static function trimCamel (str:String):String {
    86  		var o:String = new String ();
    87  		for (var i = 0; i < str.length; i++) {
    88  			if (str.charAt (i) != " ") {

    90  					o += str.charAt (i).toUpperCase ();
    91  					justPassedSpace = false;

   120  	/**
   121  	 * encode html
   122  	 * @param str (String)
   123  	 * @return String
   124  	 */
   125  	public static function htmlEncode (str:String):String {
   126  		var s:String = str, a = new String ();
   127  		a = s.split ("&"), s = a.join ("&amp;");
   128  		a = s.split (" "), s = a.join ("&nbsp;");
   129  		a = s.split ("<"), s = a.join ("&lt;");
   130  		a = s.split (">"), s = a.join ("&gt;");
   131  		a = s.split ("\""), s = a.join ("&quot;");
   132  		return s;
   133  	}

   135  	/**
   136  	 * strip html markup tags
   137  	 * @param str (String)
   138  	 * @return String
   139  	 */
   140  	public static function stripTags (str:String):String {
   141  		var s:Array = new Array ();
   142  		var c:Array = new Array ();
   143  		for (var i = 0; i < str.length; i++) {
   144  			if (str.charAt (i) == "<") {
   145  				s.push (i);
   146  			} else if (str.charAt (i) == ">") {
   147  				c.push (i);
   148  			}
   149  		}
   150  		var o:String = str.substring (0, s[0]);
   151  		for (var j = 0; j < c.length; j++) {
   152  			o += str.substring (c[j] + 1, s[j + 1]);
   153  		}
   154  		return o;
   155  	}

   166  	/**

   168  	 * @param str (String)

   171  	public static function toDoubleQuote (str:String):String {
   172  		var sq:String = "'";
   173  		var dq:String = String.fromCharCode (34);
   174  		return str.split (sq).join (dq);

   188  	/**
   189  	 * Remove all formatting and return cleaned numbers from string.
   190  	 * @param str (String)
   191  	 * @return String
   192  	 * {@code Usage: 
   193  	 * 	StringUtils.toNumeric("123-123-1234"); // returns 1221231234 
   194  	 * }
   195  	 */
   196  	public static function toNumeric (str:String):String {
   197  		var len:Number = str.length;
   198  		var result:String = "";
   199  		for (var i:Number = 0; i < len; i++) {
   200  			var code:Number = str.charCodeAt(i);
   201  			if (code >= 48 && code <= 57) {
   202  				result += str.substr(i, 1);
   203  			}
   204  		}
   205  		return result;
   206  	}
