     9   import com.sekati.display.IBaseClip;
    10   import com.sekati.core.KeyFactory;
    11   import com.sekati.reflect.Stringifier;
    12   import com.sekati.utils.MovieClipUtils;

    14  /**
    15   * This is the foundational MovieClip class and should be
    16   * thought of as the main building block of the SASAPI framework.
    17   * @see {@link com.sekati.display.CoreClip}
    18   */
    19  class com.sekati.display.BaseClip extends MovieClip implements IBaseClip {
    20  	
    21  	private var _this:MovieClip;
    22  	private var __isClean:Boolean;
    23  	
    24  	public function BaseClip() {
    25  		_this = this;
    26  		__isClean = false;
    27  		KeyFactory.inject(_this);
    28  	}
    29  	
    30  	/**
    31  	 * Destroy object elements and events for proper garbage collection.
    32  	 * This is a generic destroy method to insure that, at a minimum, the 
    33  	 * enterFrame and clip itself are removed when destroy is called. 
    34  	 * This behavior can be overwritten by BaseClip's subclasses. 
    35  	 * If you wish to use BaseClip's destroy in addition to your subclass 
    36  	 * destroy you may do so via: 
    37  	 * {@code
    38  	 * 	super.destroy(); 
    39  	 * }
    40  	 */	
    41  	public function destroy():Void {
    42  		__isClean = true;
    43  		_this.onEnterFrame = null;
    44  		for(var i in _this) {
    45  			MovieClipUtils.rmClip(_this[i]);	
    46  		}
    47  		MovieClipUtils.rmClip(_this);		
    48  	}
    49  
    50  	/**
    51  	 * if destroy hasnt already been called manually
    52  	 * run it onUnload.
    53  	 */
    54  	public function onUnload():Void {
    55  		if(!__isClean) {
    56  			destroy();
    57  		}	
    58  	}
    59  
    60  	/**
    61  	 * return reflective output
    62  	 * @return String
    63  	 */	
    64  	public function toString():String {
    65  		return Stringifier.stringify(this);
    66  	}
    67  }
