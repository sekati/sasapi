     9   import com.sekati.display.IBaseClip;
    10   import com.sekati.display.ITweenClip;
    11   import com.sekati.core.KeyFactory;
    12   import com.sekati.reflect.Stringifier;

    16  /**
    17   * This is the foundational MovieClip class and should be
    18   * thought of as the main building block of the SASAPI framework.
    19   * @see {@link com.sekati.display.CoreClip}
    20   */
    21  class com.sekati.display.BaseClip extends MovieClip implements IBaseClip, ITweenClip {
    22  
    23  	private var _this:MovieClip;
    24  	private var __isClean:Boolean;
    25  	
    26  	public function BaseClip() {
    27  		_this = this;
    28  		__isClean = false;
    29  		KeyFactory.inject(_this);
    30  	}
    31  
    32  	/**
    33  	 * if destroy hasnt already been called manually
    34  	 * run it onUnload.
    35  	 */
    36  	public function onUnload():Void {
    37  		if(!__isClean) {
    38  			destroy();
    39  		}	
    40  	}
    41  	
    42  	/**
    43  	 * Destroy object elements and events for proper garbage collection.
    44  	 * This is a generic destroy method to insure that, at a minimum, the 
    45  	 * enterFrame and clip itself are removed when destroy is called. 
    46  	 * This behavior can be overwritten by BaseClip's subclasses. 
    47  	 * If you wish to use BaseClip's destroy in addition to your subclass 
    48  	 * destroy you may do so via: 
    49  	 * {@code
    50  	 * 	super.destroy(); 
    51  	 * }
    52  	 */	
    53  	public function destroy():Void {
    54  		__isClean = true;
    55  		_this.onEnterFrame = null;
    56  		for(var i in _this) {
    57  			MovieClipUtils.rmClip(_this[i]);	
    58  		}
    59  		MovieClipUtils.rmClip(_this);		
    60  	}
    61  	
    62  	/**
    63  	 * A very basic wrapper to apply a tween to the current object via {@link com.sekati.transitions.Tweener.addTween}
    64  	 * @see {@link http://hosted.zeh.com.br/tweener/docs/en-us/}
    65  	 * @param tweenerObject (Object)
    66  	 * @return Boolean
    67  	 */
    68  	public function tween (tweenerObject:Object):Void {
    69  		Tweener.addTween (_this, tweenerObject);
    70  	}	
    71  
    72  	/**
    73  	 * return reflective output
    74  	 * @return String
    75  	 */	
    76  	public function toString():String {
    77  		return Stringifier.stringify(this);
    78  	}
    79  }
