     9   import com.sekati.display.UIClip;

    14  /**
    15   * LiquidClip - mixin for any subclass which needs to respond to {@link com.lti.display.StageDisplay} events. 
    16   * Extends {@link com.lti.display.UIClip} to allow for easy/automatic TextField CSS styling.
    17   */
    18  class com.sekati.display.LiquidClip extends UIClip {
    19  	
    20  	/**
    21  	 * Constructor
    22  	 */
    23  	public function LiquidClip() {
    24  		super();
    25  		Dispatcher.$.addEventListener(StageDisplay.onStageResizeEVENT, Delegate.create (_this, _onResize));
    26  		Dispatcher.$.addEventListener(StageDisplay.onStageResizeCompleteEVENT, Delegate.create (_this, _onResizeComplete));
    27  	}
    28  
    29  	/**
    30  	 * Clip has been loaded and registered on Stage.
    31  	 * @return Void
    32  	 */
    33  	public function configUI():Void {
    34  		super.configUI();
    35  		_onResize();
    36  		_onResizeComplete();
    37  	}
    38  	
    39  	/**
    40  	 * Application has been Configured (Config & Data loaded).
    41  	 * NOTE: Automatically applies the application stylesheet to all TextField Objects!
    42  	 * @return Void
    43  	 */
    44  	public function onAppConfigured():Void {
    45  		super.onAppConfigured();
    46  	}
    47  	
    48  	/**
    49  	 * _onResize stub: fires when Stage resize has occurs.
    50  	 * @return Void
    51  	 */
    52  	public function _onResize():Void {
    53  		//App.log.info(_this, StageDisplay.onStageResizeEVENT+" Fired!");
    54  	}
    55  	
    56  	/**
    57  	 * _onResizeComplete stub: fires {@link App.STAGE_RESIZE_DELAY} milliseconds after a Stage resize has occured.
    58  	 * @return Void
    59  	 */	
    60  	public function _onResizeComplete():Void {
    61  		//App.log.info(_this, StageDisplay.onStageResizeCompleteEVENT+" Fired!");
    62  	}
    63  	
    64  	/**
    65  	 * Remove Dispatcher listeners onUnload.
    66  	 * @return Void
    67  	 */
    68  	public function onUnload():Void {
    69  		super.onUnload();
    70  		Dispatcher.$.removeEventListener(StageDisplay.onStageResizeEVENT, _this);	
    71  		Dispatcher.$.removeEventListener(StageDisplay.onStageResizeCompleteEVENT, _this);
    72  	}	
    73  	
    74  }
