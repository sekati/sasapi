    18  	/**
    19  	 * Random flip numbers to anim text
    20  	 * @param tf (TextField)
    21  	 * @param str (String)
    22  	 * @param maxPass (Number)
    23  	 */
    24  	public static function flipNum (tf:TextField, str:String, maxPass:Number):Void {
    25  		tf.html = true;
    26  		tf.autoSize = true;
    27  		var s:String = StringUtils.stripTags(str);
    28  		var randString:String;
    29  		var curLetter:Number = 0;
    30  		var passCount:Number = 0;
    31  		var rLength:Number = Math.ceil(s.length/4);
    32  		if (rLength < 3) rLength = 3;
    33  		var anim:MovieClip = AnimHandler.create(tf);
    34  		anim.onEnterFrame = function ():Void {
    35  			if(passCount == maxPass) {
    36  				passCount = 0;
    37  				tf.htmlText = s.substring(0, curLetter);
    38  				curLetter++;
    39  			} else {
    40  				randString = (s.length-curLetter > rLength) ? StringUtils.randNum(rLength) : StringUtils.randNum(s.length - curLetter);
    41  				tf.htmlText = s.substring(0, curLetter) + randString;
    42  				passCount++;
    43  			}
    44  			if (curLetter > s.length) {
    45  				anim.onEnterFrame = null;
    46  				tf.htmlText = str;
    47  				trace("fin"); 
    48  			}			
    49  		};
    50  	}

    52  	/**
    53  	 * Easing alpha type in text animation
    54  	 * @param tf (TextField) target text
    55  	 * @param str (String) string to animate in
    56  	 * @param cb (Function) callback
    57  	 * @return Void
    58  	 */
    59  	public static function atype (tf:TextField, str:String, cb:Function):Void {
    60  		tf.html = true;
    61  		tf._alpha = 0;
    62  		var anim:MovieClip = AnimHandler.create(tf);
    63  		anim.cursor = 0;
    64  		anim.dLen = StringUtils.stripTags (str).length;
    65  		anim.onEnterFrame = function():Void {
    66  			if (anim.dLen > 30) {
    67  				anim.cursor += 16 + random (16);
    68  			} else if (anim.dLen > 20) {
    69  				anim.cursor += 3 + random (4);
    70  			} else {
    71  				anim.cursor += 1;
    72  			}
    73  			tf.htmlText = str.substr (0, anim.cursor);
    74  			anim.dLen = StringUtils.stripTags (str).length - tf.text.length;
    75  			tf._alpha = int (tf.text.length * 100 / str.length);
    76  			//
    77  			if (anim.dLen <= 1) {
    78  				anim.destroy();
    79  				tf.htmlText = str;
    80  				tf._alpha = 100;
    81  				cb ();
    82  			}
    83  		};
    84  	}

    86  	/**
    87  	 * easing alpha type out  text animation
    88  	 * @param tf (TextField) target text
    89  	 * @param cb (Function) callback
    90  	 * @return Void
    91  	 */
    92  	public static function uatype (tf:TextField, cb:Function):Void {
    93  		var sLen:Number = tf.htmlText.length;
    94  		trace ("uatype " + sLen);
    95  		// create anim handler                
    96  		var anim:MovieClip = AnimHandler.create(tf);
    97  		anim.cursor = tf.htmlText.length;
    98  		anim.onEnterFrame = function():Void {
    99  			if (anim.cursor == 0) {
   100  				//trace ("uatype done");
   101  				anim.destroy();
   102  				tf._alpha = 100;
   103  				if (cb) {
   104  					cb ();
   105  				}
   106  			} else {
   107  				//trace ("running: " + anim.cursor);
   108  				anim.cursor -= (40 + random (25));
   109  				tf._alpha = int (tf.htmlText.length * 100 / sLen);
   110  				if (anim.cursor < 0) {
   111  					anim.cursor = 0;
   112  					tf._alpha = 0;
   113  				}
   114  				tf.htmlText = tf.htmlText.substr (0, anim.cursor);
   115  			}
   116  		};
   117  	}

   119  	/**
   120  	 * text animation pushes text into view
   121  	 * @param tf (TextField) target text
   122  	 * @param str (String) string to animate in
   123  	 * @param ms (Number) miliseconds
   124  	 * @param htm (Boolean) support htmlText
   125  	 * @param cb (Function) callback
   126  	 * @return Void
   127  	 */
   128  	public static function pushtype (tf:TextField, str:String, ms:Number, htm:Boolean, cb:Function):Void {
   129  		if (htm == true) {
   130  			var hstr:String = str;
   131  			str = StringUtils.stripTags (str);
   132  			tf.html = true;
   133  		}
   134  		if (tf.pushing != true) {
   135  			tf.pushing = true;
   136  			var from:Number = str.length;
   137  			tf.wordWrap = true;
   138  			var cp:Number = (str.length > 250) ? 26 : Math.floor (str.length / 4);
   139  			if (cp < 1) {
   140  				cp = 1;
   141  			}
   142  			var myInterval:Number = setInterval (function():Void {
   143  				if (from > 0) {
   144  					from -= cp;
   145  					var to:Number = str.length;
   146  					tf.text = str.substring (from, to);
   147  				}
   148  				if (tf.text.length == str.length) {
   149  					clearInterval (myInterval);
   150  					if (htm == true) {
   151  						tf.htmlText = hstr;
   152  					}
   153  					tf.pushing = undefined;
   154  					cb ();
   155  				}
   156  				updateAfterEvent ();
   157  			}, ms);
   158  		}
   159  	}

   161  	/**
   162  	 * text animation "types" text into field
   163  	 * @param tf (TextField) target text
   164  	 * @param str (String) string to animate in
   165  	 * @param cursor (String) cursor character
   166  	 * @param isFocus (Boolean) switch textfield to input type and add caret to end of string
   167  	 * @param alphaStep (Number) alpha fade increment
   168  	 * @param pi (Number) pause intervals
   169  	 * @param ms (Number) miliseconds
   170  	 * @param cb (Function) callback
   171  	 * @return Void
   172  	 */
   173  	public static function itype (tf:TextField, str:String, cursor:String, isFocus:Boolean, alphaStep:Number, pi:Number, ms:Number, cb:Function):Void {
   174  		var anim:String = "$__xtype__", fc:Number = 0, index:Number = 0, dLen:Number = str.length + 1;
   175  		if (tf[anim]) {
   176  			clearInterval (tf[anim]);
   177  		}
   178  		delete tf[anim];
   179  		tf.autoSize = true, tf._alpha = 0, tf.text = "", tf.type = "dynamic";
   180  		tf[anim] = setInterval (function ():Void {
   181  			if (fc >= pi) {
   182  				if (tf._alpha < 100) {
   183  					tf._alpha += alphaStep;
   184  				}
   185  				if (dLen > 26) {
   186  					index += 4 + random (3);
   187  				} else if (dLen > 10) {
   188  					index += 2 + random (2);
   189  				} else {
   190  					index += 1;
   191  				}
   192  				tf.text = str.substr (0, index) + cursor;
   193  				dLen = (str.length + 1) - tf.length;
   194  				if (dLen < 2) {
   195  					clearInterval (tf[anim]);
   196  					delete tf[anim];
   197  					if (isFocus) {
   198  						tf.type = "input";
   199  						TextUtils.caret (tf, str.length);
   200  					}
   201  					tf.text = str;
   202  					tf._alpha = 100;
   203  					cb ();
   204  				}
   205  			} else {
   206  				fc++;
   207  			}
   208  			updateAfterEvent ();
   209  		}, ms);
   210  	}

   212  	/**
   213  	 *  text animation "types" text out of field
   214  	 * @param tf (TextField) target text
   215  	 * @param cursor (String) cursor character
   216  	 * @param isFocus (Boolean) switch textfield to input type and add caret to end of string
   217  	 * @param ms (Number) miliseconds
   218  	 * @param cb (Function) callback
   219  	 * @return Void
   220  	 */
   221  	public static function otype (tf:TextField, cursor:String, isFocus:Boolean, ms:Number, cb:Function):Void {
   222  		var str:String = tf.text, anim:String = "$__xtype__", undex:Number = 0, toc:Number = str.length + 1;
   223  		tf.autoSize = true, tf.type = "dynamic";
   224  		if (tf[anim]) {
   225  			clearInterval (tf[anim]);
   226  		}
   227  		delete tf[anim];
   228  		tf[anim] = setInterval (function ():Void {
   229  			if (toc > str.length) {
   230  				if (toc > 26) {
   231  					undex -= 4 + random (3);
   232  				} else if (toc > 10) {
   233  					undex -= 2 + random (2);
   234  				} else {
   235  					undex -= 1;
   236  				}
   237  				// old: if(toc>26) undex-=1; else if(toc>10) undex-=2+random(2); else undex-=4+random(3);
   238  				tf.text = str.substr (0, undex) + cursor;
   239  				toc = (str.length - 1) + tf.length;
   240  			} else {
   241  				clearInterval (tf[anim]);
   242  				delete tf[anim];
   243  				tf.text = "";
   244  				if (isFocus) {
   245  					tf.type = "input";
   246  					TextUtils.caret (tf, 0);
   247  				}
   248  				cb ();
   249  			}
   250  			updateAfterEvent ();
   251  		}, ms);
   252  	}

   254  	/**
   255  	 * text animation type text in, pauses and types text out of  field
   256  	 * @param tf (TextField) target text
   257  	 * @param str (String) string to animate in
   258  	 * @param cursor (String) cursor character
   259  	 * @param isFocus (Boolean) switch textfield to input type and add caret to end of string
   260  	 * @param alphaStep (Number) alpha fade increment
   261  	 * @param pi (Number) pause intervals
   262  	 * @param ms (Number) miliseconds
   263  	 * @param cb (Function) callback
   264  	 * @return Void
   265  	 */
   266  	public static function iotype (tf:TextField, str:String, cursor:String, isFocus:Boolean, alphaStep:Number, pauseIndex:Number, ms:Number, cb:Function):Void {
   267  		var anim:String = "$__iotype__", fc:Number = 0, index:Number = 0, undex:Number = 0, dLen:Number = str.length + 1, toc:Number = str.length + 1;
   268  		if (tf[anim]) {
   269  			clearInterval (tf[anim]);
   270  		}
   271  		delete tf[anim];
   272  		tf.autoSize = true, tf._alpha = 0, tf.text = "", tf.type = "dynamic";
   273  		tf[anim] = setInterval (function ():Void {
   274  			if (tf._alpha < 100) {
   275  				tf._alpha += alphaStep;
   276  			}
   277  			if (dLen > 26) {
   278  				index += 4 + random (3);
   279  			} else if (dLen > 10) {
   280  				index += 2 + random (2);
   281  			} else {
   282  				index += 1;
   283  			}
   284  			tf.text = str.substr (0, index) + cursor;
   285  			dLen = (str.length + 1) - tf.length;
   286  			if (dLen < 2) {
   287  				// finished type-in
   288  				tf.text = str;
   289  				tf._alpha = 100;
   290  				if (fc >= pauseIndex) {
   291  					// if pause is finished
   292  					//--------------------------------------
   293  					if (toc > str.length) {
   294  						//--untype start--
   295  						//if(toc>26) undex -= 4+random(3); else if(toc>10) undex -= 2+random(2); else undex -= 1; // old
   296  						if (toc > 26) {
   297  							undex -= 1;
   298  						} else if (toc > 10) {
   299  							undex -= 2 + random (2);
   300  						} else {
   301  							undex -= 4 + random (3);
   302  						}
   303  						tf.text = str.substr (0, undex) + cursor;
   304  						toc = (str.length - 1) + tf.length;
   305  						//--untype stop--
   306  					} else {
   307  						// finished type-out: exit
   308  						clearInterval (tf[anim]);
   309  						delete tf[anim];
   310  						tf.text = "";
   311  						if (isFocus) {
   312  							tf.type = "input";
   313  							TextUtils.caret (tf, str.length);
   314  						}
   315  						cb ();
   316  					}
   317  					//--------------------------------------
   318  				} else {
   319  					fc++;
   320  				}
   321  			}
   322  			updateAfterEvent ();
   323  		}, ms);
   324  	}
