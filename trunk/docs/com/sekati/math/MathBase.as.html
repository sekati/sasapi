    11   /**
    12    * static class wrapping various Math utilities
    13    */
    14  class com.sekati.math.MathBase {

    16  	/**
    17  	 * Returns the highest value of all passed arguments
    18  	 * Like Math.max() but supports any number of args passed to it
    19  	 */
    20  	public static function max ():Number {
    21  		return maxArray (arguments);
    22  	}
    23  
    24  	/**
    25  	 * Returns the lowest value of all passed arguments
    26  	 * Like Math.min() but supports any number of args passed to it
    27  	 */
    28  	public static function min ():Number {
    29  		return minArray (arguments);
    30  	}
    31  
    32  	/**
    33  	 * Returns the highest value of all items in array
    34  	 * Like Math.max() but supports any number of items
    35  	 */
    36  	public static function maxArray (a:Array):Number {
    37  		var val:Number = null;
    38  		for (var i in a) {
    39  			if (a[i] > val || val == null) {
    40  				val = Number (a[i]);
    41  			}
    42  		}
    43  		return val;
    44  	}
    45  
    46  	/**
    47  	 * Returns the lowest value of all items in array
    48  	 * Like Math.min() but supports any number of items
    49  	 */
    50  	public static function minArray (a:Array):Number {
    51  		var val:Number = null;
    52  		for (var i in a) {
    53  			if (a[i] < val || val == null) {
    54  				val = Number (a[i]);
    55  			}
    56  		}
    57  		return val;
    58  	}
    59  
    60  	/**
    61  	 * Same as Math.foor with extra argument to specify number of decimals
    62  	 */
    63  	public static function floor(val:Number, decimal:Number):Number {
    64  		var n:Number = Math.pow(10,decimal);
    65  		return Math.floor(val*n)/n;
    66  	}	
    67  
    68  	/**
    69  	 * Round to a given amount of decimals
    70  	 */
    71  	public static function round (val:Number, decimal:Number):Number {
    72  		return Math.round (val * Math.pow (10, decimal)) / Math.pow (10, decimal);
    73  	}
    74  
    75  	/**
    76  	 * Round to nearest .5
    77  	 * {@code Example:
    78  	 * 	trace(MathBase.roundHalf(4.47)); // returns 4.5
    79  	 * }
    80  	 */	
    81  	public static function roundHalf (val:Number):Number {
    82  		var num:String = String (Math.round (val * 10) / 10);
    83  		var tmp:Array = num.split (".");
    84  		var integer:String = tmp[0];
    85  		var decimal:Number = tmp[1];
    86  		if (decimal >= 3 && decimal <= 7 && decimal != null) {
    87  			decimal = 5;
    88  		} else {
    89  			if (decimal > 7) {
    90  				integer = Number (integer) + 1;
    91  			}
    92  			decimal = 0;
    93  		}
    94  		return Number (integer + "." + decimal);
    95  	}
    96  
    97  	/**
    98  	 * Will constrain a value to the defined boundaries
    99  	 * {@code Examples:
   100  	 * val: 20, 2 to 5    this will give back 5 since 5 is the top boundary
   101  	 * val: 3, 2 to 5     this will give back 3
   102  	 * }
   103  	 */
   104  	public static function constrain (val:Number, min:Number, max:Number):Number {
   105  		if (val < min) {
   106  			val = min;
   107  		} else if (val > max) {
   108  			val = max;
   109  		}
   110  		return val;
   111  	}
   112  	
   113  	/**
   114  	 * Return the proportional value of two pairs of numbers.
   115  	 * @param x1 (Number)
   116  	 * @param x2 (Number)
   117  	 * @param y1 (Number)
   118  	 * @param y2 (Number)
   119  	 * @param x (Number) optional
   120  	 * @return Number
   121  	 */
   122  	public static function proportion (x1:Number, x2:Number, y1:Number, y2:Number, x:Number):Number {
   123  		var n:Number = (!x) ? 1 : x;
   124  		var slope:Number = (y2 - y1) / (x2 - x1);
   125  		return (slope * (n - x1) + y1);
   126  	} 
   127  
   128  	/**
   129  	 * Check if number is Odd (convert to Integer if necessary)
   130  	 */
   131  	public static function isOdd (n:Number):Boolean {
   132  		var int:Integer = new Integer(n);
   133  		return Boolean(int%2);	
   134  	}
   135  
   136  	/**
   137  	 * Check if number is Even (convert to Integer if necessary)
   138  	 */
   139  	public static function isEven (n:Number):Boolean {
   140  		var int:Integer = new Integer(n);
   141  		return (int%2 == 0);
   142  	}
   143  
   144  	/**
   145  	 * Check if number is an Integer
   146  	 */
   147  	public static function isInteger (n:Number):Boolean {
   148  		return (n%1 == 0);
   149  	}
   150  
   151  	/**
   152  	 * Check if number is Natural (positive Integer)
   153  	 */
   154  	public static function isNatural(n:Number):Boolean {
   155  		return (n >= 0 && n%1 == 0);
   156  	}

   159  	// RANDOMS
   160  	/**
   161  	 * Returns a random number inside a specific range
   162  	 */	
   163  	public static function rnd(start:Number, end:Number):Number {
   164  		return Math.round( Math.random()*(end-start))+start;
   165  	}
   166  
   167  	/**
   168  	 * Returns a set of random numbers inside a specific range (unique numbers is optional)
   169  	 */
   170  	public static function inRangeSet (min:Number, max:Number, count:Number, unique:Boolean):Array {
   171  		var rnds:Array = new Array ();
   172  		if (unique && count <= max - min + 1) {
   173  			//unique
   174  			// create num range array
   175  			var nums:Array = new Array ();
   176  			for (var i = min; i <= max; i++) {
   177  				nums.push (i);
   178  			}
   179  			for (var i = 1; i <= count; i++) {
   180  				// random number
   181  				var rn = Math.floor (Math.random () * nums.length);
   182  				rnds.push (nums[rn]);
   183  				nums.splice (rn, 1);
   184  			}
   185  		} else {
   186  			//non unique
   187  			for (var i = 1; i <= count; i++) {
   188  				rnds.push (randRangeInt (min, max));
   189  			}
   190  		}
   191  		return rnds;
   192  	}
   193  
   194  	/**
   195  	 * Returns a random float number within a given range
   196  	 */
   197  	public static function randRangeFloat (min:Number, max:Number):Number {
   198  		return Math.random () * (max - min) + min;
   199  	}
   200  
   201  	/**
   202  	 * Returns a random int number within a given range
   203  	 */
   204  	public static function randRangeInt (min:Number, max:Number):Number {
   205  		return Math.floor (Math.random () * (max - min + 1) + min);
   206  	}

   209  	}
