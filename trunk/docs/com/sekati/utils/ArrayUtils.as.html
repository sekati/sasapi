     9   /**
    10    * static class wrapping various Array utilities
    11    */
    12  class com.sekati.utils.ArrayUtils {
    13  	/**
    14  	 * shuffle array items
    15  	 * @param a (Array)
    16  	 * @return Void
    17  	 */
    18  	public static function shuffle (a:Array):Void {
    19  		for (var i:Number = 0; i < a.length; i++) {
    20  			var tmp:Object = a[i];
    21  			var randomNum:Number = random (a.length);
    22  			a[i] = a[randomNum];
    23  			a[randomNum] = tmp;
    24  		}
    25  	}
    26  	/**
    27  	 * return alphabetically sorted array
    28  	 * @param a (Array)
    29  	 * @return Array
    30  	 */
    31  	public static function asort (a:Array):Array {
    32  		var aFn = function (element1:String, element2:String) {
    33  			var element1:String = element1.toUpperCase ();
    34  			var element2:String = element2.toUpperCase ();
    35  			return element1 > element2;
    36  		};
    37  		return a.sort (aFn);
    38  	}
    39  	/**
    40  	 * return array with duplicate entries removed
    41  	 * @param a (Array)
    42  	 * @return Array
    43  	 */
    44  	public static function removeDuplicate (a:Array):Array {
    45  		a.sort ();
    46  		var o:Array = new Array ();
    47  		for (var i:Number = 0; i < a.length; i++) {
    48  			if (a[i] != a[i + 1]) {
    49  				o.push (a[i]);
    50  			}
    51  		}
    52  		return o;
    53  	}
    54  	/**
    55  	 * compare two arrays for a matching value
    56  	 * @param a1 (Array)
    57  	 * @param a2 (Array)	
    58  	 * @return Boolean
    59  	 */	
    60  	public static function matchValues (a1:Array, a2:Array):Boolean {
    61  		for (var f:Number = 0; f < a1.length; f++) {
    62  			for (var l:Number = 0; l < a2.length; l++) {
    63  				if (a2[l].toLowerCase () === a1[f].toLowerCase ()) {
    64  					return true;
    65  				}
    66  			}
    67  		}
    68  		return false;
    69  	}
    70  }
