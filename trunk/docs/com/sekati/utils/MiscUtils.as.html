   116  	/**
   117  	 * convert fahrenheit to celsius
   118  	 * @param f (Number) fahrenheit value
   119  	 * @param p (Number) number of decimal after int '.' without round
   120  	 * @return Number
   121  	 */
   122  	public static function f2c(f:Number, p:Number):Number {
   123  		var d;
   124  		var r = (5 / 9) * (f - 32);
   125  		var s = r.toString ().split (".");
   126  		if (s[1] != undefined) {
   127  			d = s[1].substr (0, p);
   128  		} else {
   129  			var i = p;
   130  			while (i > 0) {
   131  				d += "0";
   132  				i--;
   133  			}
   134  		}
   135  		var c = s[0] + "." + d;
   136  		return Number (c);		
   137  	}
