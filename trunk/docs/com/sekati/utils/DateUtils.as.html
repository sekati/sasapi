     9   /**
    10    * Static function for handling dates from db and converting them into readable strings
    11    * 
    12    * {@code Usage:
    13    * var d = DateUtils.dateFromDB("2006-06-01 12:10:45");
    14    * trace(DateUtils.days[d.getDay()] + ", " + DateUtils.months[d.getMonth()] + " " + d.getDate() + ", " + (d.getHours()%12) + ":" + DateUtils.padTime(d.getMinutes()) + ((d.getHours() > 12) ? "pm" :"am"));
    15    * }
    16    */
    17  class com.sekati.utils.DateUtils {
    18  
    19  	/**
    20  	 * gets 0 indexed array of months for use with Date.getMonth()
    21  	 */
    22  	public static function get months ():Array {
    23  		return new Array ("January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November", "December");
    24  	}
    25  	
    26  	/**
    27  	 * gets 0 indexed array of short months for use with Date.getMonth()
    28  	 */
    29  	public static function get smonths ():Array {
    30  		return new Array ("Jan","Feb","Mar","Apr","May","Jun","Jul","Aug","Sep","Oct","Nov","Dec");	
    31  	}
    32  
    33  	/**
    34  	 * gets 0 indexed array of days for use with Date.getDay()
    35  	 */
    36  	public static function get days ():Array {
    37  		return new Array ("Sunday", "Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday");
    38  	}
    39  	
    40  	/**
    41  	 * get 0 indexed array of days for use with Date.getDay();
    42  	 */
    43  	public static function get sdays ():Array {

    45  	}
    46  
    47  	/**
    48  	 * pads hours,minutes or seconds with a leading 0, 12:01 doesn't end up 12:1
    49  	 */
    50  	public static function padTime (inn):String {
    51  		return (String (inn).length < 2) ? ("0" + inn) : inn;
    52  	}
    53  
    54  	/**
    55  	 * converts a db formatted date string into a Flash Date object
    56  	 * @param dbDate (String) date in YYYY-MM-DD HH:MM:SS format
    57  	 * @return Date
    58  	 */
    59  	public static function dateFromDB (dbdate):Date {
    60  		var tmp = dbdate.split (" ");
    61  		var dates = tmp[0].split ("-");
    62  		var hours = tmp[1].split (":");
    63  		var d = new Date (dates[0], dates[1] - 1, dates[2], hours[0], hours[1], hours[2]);
    64  		return d;
    65  	}
    66  
    67  	/**
    68  	 * takes 24hr hours and converts to 12 hour with am/pm
    69  	 * @param hour24 (Number)
    70  	 * @return Object
    71  	 */
    72  	public static function getHoursAmPm (hour24:Number):Object {
    73  		var returnObj:Object = new Object ();
    74  		returnObj.ampm = (hour24 < 12) ? "am" : "pm";
    75  		var hour12:Number = hour24 % 12;
    76  		if (hour12 == 0) {
    77  			hour12 = 12;
    78  		}
    79  		returnObj.hours = hour12;
    80  		return returnObj;
    81  	}
    82

    84  	}
    85  }
