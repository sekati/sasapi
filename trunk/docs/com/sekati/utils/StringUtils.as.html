    54  	/**
    55  	 * remove spaces at end and beginning of the string only
    56  	 * @param str (String)
    57  	 * @return String
    58  	 */
    59  	public static function trim (str:String):String {
    60  		var index0:Number = 0;
    61  		while (str.charAt (index0) == " ") {
    62  			index0++;
    63  		}
    64  		var index1:Number = str.length - 1;
    65  		while (str.charAt (index1) == " ") {
    66  			index1--;
    67  		}
    68  		return str.substring (index0, index1 + 1);
    69  	}

    71  	/**
    72  	 * remove spaces  tabs, line feeds, carrige returns from string
    73  	 * @param str (String)
    74  	 * @return String
    75  	 */
    76  	public static function xtrim (str:String):String {
    77  		var o:String = new String ();
    78  		var TAB:Number = 9;
    79  		var LINEFEED:Number = 10;
    80  		var CARRIAGE:Number = 13;
    81  		var SPACE:Number = 32;
    82  		for (var i:Number = 0; i < str.length; i++) {
    83  			if (str.charCodeAt (i) != SPACE && str.charCodeAt (i) != CARRIAGE && str.charCodeAt (i) != LINEFEED && str.charCodeAt (i) != TAB) {
    84  				o += str.charAt (i);
    85  			}
    86  		}
    87  		return o;
    88  	}

   121  	/**
   122  	 * encode html
   123  	 * @param str (String)
   124  	 * @return String
   125  	 */
   126  	public static function htmlEncode (str:String):String {
   127  		var s:String = str, a:Object = new String();
   128  		a = s.split ("&"), s = a.join ("&amp;");
   129  		a = s.split (" "), s = a.join ("&nbsp;");
   130  		a = s.split ("<"), s = a.join ("&lt;");
   131  		a = s.split (">"), s = a.join ("&gt;");
   132  		a = s.split ("\""), s = a.join ("&quot;");
   133  		return s;
   134  	}

   141  	public static function htmlDecode(t:String):String {
   142  		t = t.split("&reg;").join("¨");
   143  		t = t.split("&copy;").join("©");
   144  		t = t.split("&rsquo;").join("'");
   145  		t = t.split("&ldquo;").join("\"");
   146  		t = t.split("&rdquo;").join("\"");
   147  		t = t.split("&hellip;").join("...");
   148  		t = t.split("&middot;").join("*");
   149  		t = t.split("&ndash;").join("-");
   150  		t = t.split("&trade;").join("(TM)");
   151  		t = t.split("&egrave;").join();
   152  		t = t.split("&eacute;").join("Ž.");
   153  		t = t.split("&bull;").join("-");

   157  	/**
   158  	 * strip the zero off floated numbers
   159  	 * @param n (Number)
   160  	 * @return String
   161  	 */	
   162  	public static function stripZeroOnFloat(n:Number):String {
   163  		var str:String = "";
   164  		var a:Array = String(n).split(".");
   165  		if (a.length>1) {
   166  			str = (a[0] == "0") ? "."+a[1] : String(n);
   167  		} else {
   168  			str = String(n);
   169  		}
   170  		return str;
   171  	}

   173  	/**
   174  	 * strip the zero off floated numbers and remove Scientific Notation
   175  	 * @param n (Number)
   176  	 * @return String
   177  	 */
   178  	public static function stripZeroAndRepairFloat(n:Number):String {
   179  		var str:String;
   180  		var tmp:String;
   181  		var isZeroFloat:Boolean;
   182  		// +=1 to prevent scientific notation.
   183  		if(n < 1) {
   184  			tmp = String((n + 1));
   185  			isZeroFloat = true;
   186  		} else {
   187  			tmp = String(n);
   188  			isZeroFloat = false;	
   189  		}
   190  		// if we have a float strip the zero (or +=1) off!
   191  		var a:Array = tmp.split(".");
   192  		if (a.length > 1) {
   193  			str = (a[0] == "1" && isZeroFloat == true) ? "."+a[1] : tmp;
   194  		} else {
   195  			str = tmp;
   196  		}
   197  		return str;
   198  	}

   244  	/**
   245  	 * strip html markup tags
   246  	 * @param str (String)
   247  	 * @return String
   248  	 */
   249  	public static function stripTags (str:String):String {
   250  		var s:Array = new Array ();
   251  		var c:Array = new Array ();
   252  		for (var i:Number = 0; i < str.length; i++) {
   253  			if (str.charAt (i) == "<") {
   254  				s.push (i);
   255  			} else if (str.charAt (i) == ">") {
   256  				c.push (i);
   257  			}
   258  		}
   259  		var o:String = str.substring (0, s[0]);
   260  		for (var j:Number = 0; j < c.length; j++) {
   261  			o += str.substring (c[j] + 1, s[j + 1]);
   262  		}
   263  		return o;
   264  	}

   297  	/**
   298  	 * Remove all formatting and return cleaned numbers from string.
   299  	 * @param str (String)
   300  	 * @return String
   301  	 * {@code Usage: 
   302  	 * 	StringUtils.toNumeric("123-123-1234"); // returns 1221231234 
   303  	 * }
   304  	 */
   305  	public static function toNumeric (str:String):String {
   306  		var len:Number = str.length;
   307  		var result:String = "";
   308  		for (var i:Number = 0; i < len; i++) {
   309  			var code:Number = str.charCodeAt(i);
   310  			if (code >= 48 && code <= 57) {
   311  				result += str.substr(i, 1);
   312  			}
   313  		}
   314  		return result;
   315  	}
