    19  /**
    20   * Propcaster singleton creates a baseclip rectangle on stage which is used to track and broadcast
    21   * a content viewport scaled proportionally to the Stage. 
    22   * 
    23   * This wraps and simplifies the {@link com.sekati.display.StageDisplay} dispatched events and 
    24   * pushes them thru {@link com.sekati.events.Broadcaster} using the {@code onPropcast} event after 
    25   * updating the tracking clip so that the information may be used in liquid layouts that require 
    26   * positional relationships.
    27   */
    28  class com.sekati.layout.Propcaster extends CoreObject {
    29  
    30  	private static var _instance:Propcaster;
    31  	private static var _proportionW:Number = 16;
    32  	private static var _proportionH:Number = 9;
    33  	
    34  	private var _this:Propcaster;
    35  	private var _tracker:MovieClip;
    36  
    37  	/**
    38  	 * Singleton Constructor - create a tracker clip, set to proper ratio and listen for stage events.
    39  	 */
    40  	private function Propcaster() {
    41  		super();
    42  		trace("*** - Proportioncaster Initialized ...");
    43  		_this = this;
    44  		_tracker = ClassUtils.createEmptyMovieClip(com.sekati.display.BaseClip, _root, "__prop_track_rect__", {_x:0, _y:0, _alpha:25});
    45  		Rectangle.draw(_tracker, new Point(0,0), new Point(_proportionW, _proportionH), 0xFF00FF, 100, 0);
    46  		Dispatcher.$.addEventListener(StageDisplay.onStageResizeEVENT, Delegate.create (_this, update));
    47  		update();
    48  	}
    49  
    50  	/**
    51  	 * Singleton Accessor
    52  	 * @return Proportioncaster
    53  	 */
    54  	public static function getInstance():Propcaster {
    55  		if (!_instance) _instance = new Propcaster();
    56  		return _instance;
    57  	}
    58  
    59  	/**
    60  	 * shorthand singleton accessor getter
    61  	 */
    62  	 public static function get $():Propcaster {
    63  	 	return Proportioncaster.getInstance();	
    64  	 }
    65  	
    66  	/**
    67  	 * Set a new proportion to scale by.
    68  	 * @param w (Number)
    69  	 * @param h (Number)
    70  	 * @return Void
    71  	 */
    72  	public function setProp(w:Number, h:Number):Void {
    73  		_proportionW = w;
    74  		_proportionH = h;
    75  		update();
    76  	}
    77  
    78  	/**
    79  	 * Proportionally scale the tracker to fit Stage.
    80  	 * @return Void
    81  	 */
    82  	private function update():Void {
    83  		AlignUtils.scaleRatio (_tracker, Stage.width, Stage.height);
    84  		AlignUtils.alignCenter (_tracker, StageDisplay.$);
    85  		Broadcaster.$.broadcast("onPropcast", {x:x, y:y, w:w, h:h});
    86  	}
    87  	
    88  	/**
    89  	 * _x getter
    90  	 * @return Number
    91  	 */
    92  	public function get x():Number {
    93  		return _tracker._x;	
    94  	}
    95  
    96  	/**
    97  	 * _y getter
    98  	 * @return Number
    99  	 */
   100  	public function get y():Number {
   101  		return _tracker._y;	
   102  	}
   103  	
   104  	/**
   105  	 * _width getter
   106  	 * @return Number
   107  	 */	
   108  	public function get w():Number {
   109  		return _tracker._width;	
   110  	}
   111  
   112  	/**
   113  	 * _height getter
   114  	 * @return Number
   115  	 */	
   116  	public function get h():Number {
   117  		return _tracker._height;
   118  	}	
   119  }
