     8  import com.sekati.math.TrigoBase;
     9  /**
    10   * create a Point object
    11   * 
    12   * {@code Usage:
    13   * var Point = new sekati.geom.Point(10,10);
    14   * }
    15   */
    16  //
    17  class com.sekati.geom.Point {
    18  	public var x:Number;
    19  	public var y:Number;
    20  	/**
    21    	 * Constructor
    22  	 * @param nX (Number)
    23  	 * @param nY (Number)
    24  	 */
    25  	public function Point (nX:Number, nY:Number) {
    26  		x = nX;
    27  		y = nY;
    28  	}
    29  	/**
    30  	 * check if two points match
    31  	 * @param p (Point)
    32  	 * @return Boolean
    33  	 */
    34  	public function isEqual (p:Point):Boolean {
    35  		return (p.x == x && p.y == y);
    36  	}
    37  	/**
    38  	 * get distance between two points
    39  	 * @param p (Point)
    40  	 * @return Number
    41  	 */
    42  	public function getDistance (p:Point):Number {
    43  		return TrigoBase.getDistance (this, p);
    44  	}
    45  	/**
    46  	 * Algo to give the grid based distance when only vertical & horizontal moves are allowed
    47  	 * @param p (Point)
    48  	 * @return Number
    49  	 */
    50  	public function getAbsoluteGridDistance (p:Point):Number {
    51  		return Math.abs (x - p.x) + Math.abs (y - p.y);
    52  	}
    53  	/**
    54  	* Algo to give the grid based distance when diagonal moves are allowed
    55  	* by finding math.min of the differences, we're figuring out how many moves can be diagonal ones.
    56  	* Then we can just substract that number from the normal .getAbsoluteGridDistance() method since diagonals take
    57  	* 1 move instead of the usual 2
    58  	* @param p (Point)
    59  	* @return Number
    60  	*/
    61  	public function getAbsoluteGridDistanceAllowDiagonals (p:Point):Number {
    62  		var offset:Number = Math.min (Math.abs (x - p.x), Math.abs (y - p.y));
    63  		return getAbsoluteGridDistance (p) - offset;
    64  	}
    65  	/**
    66  	* Returns a new point based on this point with x and y offset values
    67  	* @param nX (Number)
    68  	* @param nY (Number)
    69  	* @return Point
    70  	*/
    71  	public function displace (nX:Number, nY:Number):Point {
    72  		return new Point (x + nX, y + nY);
    73  	}
    74  	/**
    75  	* toString rewrite for a useful trace on the instance
    76  	* @return String
    77  	*/
    78  	public function toString ():String {
    79  		return "Point:{x:" + x + ", y:" + y + "};";
    80  	}
    81  	//
    82  }
