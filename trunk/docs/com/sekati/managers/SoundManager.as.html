     8  import com.sekati.core.FWDepth;
     9  import com.sekati.math.MathBase;
    10  /**
    11   * centralizes & solves issues calling linked sounds from dll's or locally
    12   * extrapolated on ca.nectere.dllFactory.SoundFactory
    13   */
    14  class com.sekati.managers.SoundManager {
    15  	private static var _instance:SoundManager;
    16  	private var _holder:MovieClip;
    17  	private var _gSound:Sound;
    18  	private var _gMute:Boolean;
    19  	private var _gVolume:Number;
    20  	private var _gPan:Number;
    21  	/**
    22  	 * Singleton Accessor
    23  	 * @return SoundManager
    24  	 */
    25  	public static function getInstance ():SoundManager {
    26  		if (!_instance) {
    27  			_instance = new SoundManager ();
    28  		}
    29  		return _instance;
    30  	}
    31  	/**
    32  	 * shorthand singleton accessor getter
    33  	 */
    34  	 public static function get $():SoundManager {
    35  	 	return SoundManager.getInstance();	
    36  	 }	
    37  	/**
    38  	 * Singleton Private Constructor
    39  	 */
    40  	private function SoundManager () {
    41  		_holder = _root.createEmptyMovieClip ("__SoundManager__", FWDepth.SoundManager);
    42  		//no param means it'll affect the whole application
    43  		_gSound = new Sound ();
    44  		_gMute = false;
    45  	}
    46  	/**
    47  	 * returns the holder clip so we can load the DLL swf in
    48  	 * @return MovieClip
    49  	 */
    50  	public function getHolder ():MovieClip {
    51  		return _holder;
    52  	}
    53  	/**
    54  	 * returns the global volume setting
    55  	 * @return Number
    56  	 */
    57  	public function getVolume ():Number {
    58  		return _gVolume;
    59  	}
    60  	/**
    61  	 * sets the global volume
    62  	 * @param n (Number) 0 to 100
    63  	 * @return Void
    64  	 */
    65  	public function setVolume (n:Number):Void {
    66  		_gVolume = MathBase.constrain (n, 0, 100);
    67  		_gSound.setVolume (_gVolume);
    68  	}
    69  	/**
    70  	 * returns the global pan settings
    71  	 * @return Number
    72  	 */
    73  	public function getPan ():Number {
    74  		return _gPan;
    75  	}
    76  	/**
    77  	 * sets the global pan
    78  	 * @param n (Number) -100 to 100
    79  	 */
    80  	public function setPan (n:Number):Void {
    81  		_gPan = MathBase.constrain (n, -100, 100);
    82  		_gSound.setPan (_gPan);
    83  	}
    84  	/**
    85  	 * play a sound
    86  	 * @param linkage (String) linkage id for sound
    87  	 * @param loops (Number) number of times to play sound
    88  	 * @param msOffset (Number) miliseconds offset
    89  	 * @return Void
    90  	 */
    91  	public function play (linkage:String, loops:Number, msOffset:Number):Void {
    92  		if (loops == null) {
    93  			loops = 0;
    94  		}
    95  		if (msOffset == null) {
    96  			msOffset = 0;
    97  		}
    98  		var s:Sound = new Sound (_holder);
    99  		s.attachSound (linkage);
   100  		s.start (msOffset / 1000, loops);
   101  	}
   102  	/**
   103  	* toggle mute global application audio
   104  	* @param isMute (Boolean) optional force state
   105  	* @return Void
   106  	*/
   107  	public function mute (isMute:Boolean):Void {
   108  		if (!_gMute || isMute) {
   109  			_gSound.setVolume (0);
   110  			_gMute = true;
   111  		} else if (_gMute || !isMute) {
   112  			_gSound.setVolume (_gVolume);
   113  			_gMute = false;
   114  		}
   115  	}
   116  	//
   117  }
   118  // eof
