     9  /**
    10   * Context Menu management for right clicks!
    11   * {@code Usage:
    12   * 	var cm:ContextualMenu = new ContextualMenu(_level0);
    13   * 	cm.addItem("Item One", Delegate.create(this, myFunction), true, true);
    14   * 	cm.addItem("Item Two");
    15   * 	cm.removeItem("Item One"); // remove the first item
    16   * 	cm.enabled = false; // disable this menu
    17   * }
    18   */
    19  class com.sekati.ui.ContextualMenu {
    20  	
    21  	private var _target:MovieClip;
    22  	private var _isEnabled:Boolean;
    23  	private var _hasBuiltInItems:Boolean;
    24  	private var _items:Array;
    25  	private var _backup:Array;
    26  
    27  	/**
    28  	 * Custom Context Menu Constructor
    29  	 * @param target (MovieClip) object context menu should exist on
    30  	 * @return Void
    31  	 * @throws Error if no target is provided and returns without proper instantiation.
    32  	 */	
    33  	public function ContextualMenu(target:MovieClip) {
    34  		if(!target) {
    35  			throw new Error ("@@@ com.sekati.ui.ContextualMenu Error: constructor expects a 'target:MovieClip' parameter.");
    36  			return;	
    37  		}
    38  		_target = target;
    39  		_isEnabled = true;
    40  		_hasBuiltInItems = false;
    41  		_items = new Array();
    42  		_backup = new Array();
    43  	}
    44  	
    45  	/**
    46  	 * Add an item to the Context Menu
    47  	 * @param caption (String) menu item text
    48  	 * @param cb (Function) delegated callback function [optional]
    49  	 * @param isDiv (Boolean) add a divider before the item [optional, default: true]
    50  	 * @param isEnabled (Boolean) item is enabled for selection [optional, default: true]
    51  	 * @return Void
    52  	 */	 
    53  	public function addItem(caption:String, cb:Function, isDiv:Boolean, isEnabled:Boolean):Void {
    54  		cb = (!cb) ? menuClick : cb;
    55  		isDiv = (isDiv == undefined) ? true : isDiv;
    56  		isEnabled = (isEnabled == undefined) ? true : isEnabled;
    57  		_items.push({_caption:caption, _cb:cb, _isDiv:isDiv, _isEnabled:isEnabled});
    58  		buildMenu();
    59  	}
    60  
    61  	/**
    62  	 * Remove an item from the Context Menu
    63  	 * @param name (String) the caption name of the item to be removed
    64  	 * @return Boolean - true if item was sucessfully removed, false if removal failed.
    65  	 */
    66  	public function removeItem(name:String):Boolean {
    67  		for(var i:Number=0; i< _items.length; i++){
    68  			if (_items[i]._caption === name) {
    69  				_items.splice(i, 1);
    70  				buildMenu ();
    71  				return true;
    72  			}			
    73  		}
    74  		return false;
    75  	}	
    76  	
    77  	/**
    78  	 * Build the custom Context Menu.
    79  	 * @return Void
    80  	 */	
    81  	private function buildMenu():Void {
    82  		var m:ContextMenu = new ContextMenu ();
    83  		if(!_hasBuiltInItems) {
    84  			m.hideBuiltInItems ();
    85  		}
    86  		for(var i:Number=0; i<_items.length; i++) {
    87  			var cmi:ContextMenuItem = new ContextMenuItem (_items[i]._caption, _items[i]._cb, _items[i]._isDiv, _items[i]._isEnabled);
    88  			m.customItems.push(cmi);
    89  		}
    90  		_target.menu = m;
    91  	}
    92  	
    93  	/**
    94  	 * Menu enabled setter
    95  	 * @param b (Boolean) - enable (true) or disable (false) the custom Context Menu [default: true].
    96  	 * @return Void
    97  	 */
    98  	public function set enabled(b:Boolean):Void {
    99  		if (b == false) {
   100  			if (_isEnabled != false) {
   101  				_isEnabled = false;
   102  				_backup = _items;
   103  				_items = [];
   104  				buildMenu ();
   105  			}
   106  		} else {
   107  			if (_isEnabled != true) {
   108  				_isEnabled = true;
   109  				_items = _backup;
   110  				buildMenu ();
   111  			}
   112  		}
   113  	}
   114  	
   115  	/**
   116  	 * Menu enabled getter
   117  	 * @return Boolean
   118  	 */
   119  	public function get enabled():Boolean {
   120  		return _isEnabled;	
   121  	}	
   122  	
   123  	/**
   124  	 * Menu builtInItems setter
   125  	 * @param b (Boolean) - enable (true) or disable (false) the Context Menu's built in items [default: false].
   126  	 * @return Void
   127  	 */
   128  	public function set builtInItems(b:Boolean):Void {
   129  		_hasBuiltInItems = b;	
   130  	}
   131  	
   132  	/**
   133  	 * Menu builtInItems getter
   134  	 * @return Boolean
   135  	 */
   136  	public function get builtInItems():Boolean {
   137  		return _hasBuiltInItems;	
   138  	}
   139  	
   140  	/**
   141  	 * Void function to assign to items added without callback
   142  	 * @return Void
   143  	 */
   144  	public function menuClick ():Void {
   145  	}
   146  }
