     9  import com.sekati.utils.Delegate;
    10  import mx.transitions.easing.Strong;

    13   /**
    14   * Scroll handles mouseWheel, dynamic resizing of content, external size tracking 
    15   * for accordian style content scrolling, slideContent method, modal ui states, 
    16   * proportional scroller, gutter and more. 
    17   * 
    18   * {@code Usage:
    19   *  var _scroll:Scroll = new com.sekati.ui.Scroll("_y", contentMc, maskMc, gutterMc, barMc, true, true, true, true);
    20   *  var _scroll:Scroll = new com.sekati.ui.Scroll ("_y", contentMc, maskMc, scrollbar.gutter, scrollbar.bar, true, true, true, true, contentMc, .8, .5, {up:0xFFFFFF, over:0x333333});
    21   * }
    22   * 
    23   * TODO: currently scroller only repositions to accomodate content size change scrollbar 
    24   * 		 requires proportionally resizing to content size change as well (all attempts
    25   * 		 thus far produce inaccuracy/bugs. fix would remove need to re-call init upon 
    26   * 		 major content change) 
    27   */ 
    28  class com.sekati.ui.Scroll {
    29  
    30  	private var _this:Scroll;
    31  	private var _axis:String;
    32  	private var _prop:String;
    33  	private var _content:Object;
    34  	private var _gutter:MovieClip;
    35  	private var _scroller:MovieClip;
    36  	private var _mask:MovieClip;
    37  	private var _isDragged:Boolean;
    38  	private var _isResizeScroller:Boolean;
    39  	private var _isResizeGutter:Boolean;
    40  	private var _isMouseWheelEnabled:Boolean;
    41  	private var _speed:Number;
    42  	private var _contentTop:Number;
    43  	private var _contentBot:Number;
    44  	private var _contentSizeTracker:Object;
    45  	private var _gutterTop:Number;
    46  	private var _gutterBot:Number;
    47  	private var _oldPos:Number;
    48  	private var _newPos:Number;
    49  	private var _friction:Number;
    50  	private var _ratio:Number;
    51  	private var _colors:Object;
    52  	private var _slide:Tween;
    53  
    54  	/**
    55  	 * Constructor
    56  	 * @param axis (String) axis the scroller will scroll on: must be "_x" or "_y"
    57  	 * @param content (Object) content object
    58  	 * @param mask (MovieClip) content mask
    59  	 * @param gutter (MovieClip) gutter object
    60  	 * @param scroller (MovieClip) scroller bar object
    61  	 * @param isResizeGutter (Boolean) optional resize gutter to content mask
    62  	 * @param isResizeScroller (Boolean) optional resize scroller bar proportional to content
    63  	 * @param isMouseWheelEnabled (Boolean) optional enable mouseWheel scrolling (one may choose to use 2 instances of Scroll for hscroll & vscroll and only want vscroll in which case mouseWheel should only be enabled for one instance)
    64  	 * @param isInit (Boolean) optional init scroller upon instantiation, else {@link #init()} must be called manually
    65  	 * @param contentSizeTracker (Object) optional object used to track content size. Useful in accordian content where visibility or masking is used to hide partial content [default: content]
    66  	 * @param friction (Number) optional scroll motion friction [default 0.8] 
    67  	 * @param ratio (Number) optional scroll motion ratio  [default 0.5]
    68  	 * @param colors (Object) optional scroller bar modal event color settings {up:0xCCCCCC, over:0xFFFFFF}
    69  	 * @return Void
    70  	 * @throws Error if axis is not "_x" or "_y" & returns without proper instantiation
    71  	 */
    72  	public function Scroll (axis:String, content:Object, mask:MovieClip, gutter:MovieClip, scroller:MovieClip, isResizeGutter:Boolean, isResizeScroller:Boolean, isMouseWheelEnabled:Boolean, isInit:Boolean, contentSizeTracker:Object, friction:Number, ratio:Number, colors:Object) {
    73  		// scrolling axis is critical - throw an error and force breakage to gain attention
    74  		if (axis != "_x" && axis != "_y") {
    75  			throw new Error ("@@@ com.sekati.ui.Scroll Error: constructor expects axis param: '_x' or '_y'.");
    76  			return;
    77  		}
    78  		// user defined                                 
    79  		_this = this;
    80  		_axis = axis;
    81  		_content = content;
    82  		_gutter = gutter;
    83  		_scroller = scroller;
    84  		_mask = mask;
    85  		_contentSizeTracker = (!contentSizeTracker) ? content : contentSizeTracker;
    86  		_isResizeScroller = (!isResizeScroller) ? false : true;
    87  		_isResizeGutter = (!isResizeGutter) ? false : true;
    88  		_isMouseWheelEnabled = (!isMouseWheelEnabled) ? false : true;
    89  		_friction = (friction) ? friction : 0.8;
    90  		_ratio = (ratio) ? ratio : 0.5;
    91  		_colors = (colors) ? colors : {};
    92  		// class defined
    93  		_prop = (_axis == "_y") ? "_height" : "_width";
    94  		_isDragged = false;
    95  		_speed = 0;
    96  		// auto init on request ...
    97  		if (isInit) {
    98  			init ();
    99  		}
   100  	}
   101  	
   102  	/**
   103  	 * initialize scroll behavior 
   104  	 * @return Void
   105  	 */
   106  	public function init ():Void {
   107  		// define confines, mouseWheel, and set scroller to rollout color state
   108  		setConfines ();
   109  		setMouseWheel ();
   110  		scroller_onRollOut ();
   111  		// define event handlers
   112  		_scroller.onRollOver = Delegate.create (_this, scroller_onRollOver);
   113  		_scroller.onDragOver = Delegate.create (_this, scroller_onRollOver);
   114  		_scroller.onRollOut = Delegate.create (_this, scroller_onRollOut);
   115  		_scroller.onDragOut = Delegate.create (_this, scroller_onRollOut);
   116  		_scroller.onPress = Delegate.create (_this, scroller_onPress);
   117  		_scroller.onRelease = Delegate.create (_this, scroller_onRelease);
   118  		_scroller.onReleaseOutside = Delegate.create (_this, scroller_onReleaseOutside);
   119  		_scroller.onEnterFrame = Delegate.create (_this, scroller_onEnterFrame);
   120  		_gutter.onPress = Delegate.create (_this, gutter_onPress);
   121  	}
   122  	
   123  	/**
   124  	 * tween content to position, scroller will reposition accordingly
   125  	 * @param pos (Number) position to slide content on axis
   126  	 * @return Void
   127  	 */
   128  	public function slideContent (pos:Number):Void {
   129  		var newScrollPos:Number = (_contentTop - pos) * (_gutterBot - _gutterTop) / (_gutterBot - _contentBot + _scroller[_prop]) + _gutterTop;
   130  		slideScroller (newScrollPos);
   131  	}
   132  	
   133  	/**
   134  	 * tween scroller to position, content will reposition accordingly
   135  	 * @param pos (Number) position to slide scroller on axis
   136  	 * @return Void
   137  	 */
   138  	public function slideScroller (pos:Number) {
   139  		stopScroller ();
   140  		_slide = new Tween (_scroller, _axis, Strong.easeOut, _scroller[_axis], resolveScrollerPos (pos), 0.5, true);
   141  	}
   142  	
   143  	// TOOLS
   144  	
   145  	private function setConfines ():Void {
   146  		_contentTop = _content[_axis];
   147  		_gutterTop = _gutter[_axis];
   148  		_gutterBot = _gutter[_axis] + _gutter[_prop];
   149  		// scale gutter to content mask
   150  		_gutter[_prop] = (_isResizeGutter) ? _mask[_prop] : _gutter[_prop];
   151  		// set content related confines (isloated since may need to be called on size change)
   152  		updateConfines ();
   153  	}
   154  	
   155  	private function updateConfines ():Void {
   156  		_contentBot = _contentTop + _contentSizeTracker[_prop];
   157  		//_contentBot = _content[_axis] + _contentSizeTracker[_prop];
   158  		//scale scroller bar in proportion to the content to scroll
   159  		if (_isResizeScroller) {
   160  			var percent:Number = (_gutterBot - _gutterTop) / (_contentBot - _contentTop);
   161  			_scroller[_prop] = constrain (Math.round ((_gutterBot - _gutterTop) * percent), _gutterTop, _gutterBot);
   162  			//_scroller[_prop] = Math.round ((_gutterBot - _gutterTop) * percent), _gutterTop, _gutterBot;
   163  			//get new gutterBot with new size of scrollerbar                               
   164  			_gutterBot = _gutter[_axis] + _gutter[_prop] - _scroller[_prop];
   165  		}
   166  	}
   167  	
   168  	private function isScrollable ():Boolean {
   169  		// check if we need a scroller at all
   170  		var _isScrollable:Boolean = (_content[_axis] <= _mask[_axis] && _contentSizeTracker[_prop] < _mask[_prop]) ? false : true;
   171  		return _isScrollable;
   172  	}
   173  	
   174  	private function resolveScrollerPos (pos:Number) {
   175  		//makes sure scroller moves within boundaries
   176  		return Math.max (Math.min (pos, _gutterBot), _gutterTop);
   177  	}
   178  	
   179  	private function stopScroller () {
   180  		_speed = 0;
   181  		_slide.stop ();
   182  	}
   183  	
   184  	// small utils from elsewhere in com.sekati.* included to keep Scroll portable
   185  	private function constrain (val:Number, min:Number, max:Number):Number {
   186  		if (val < min) {
   187  			val = min;
   188  		} else if (val > max) {
   189  			val = max;
   190  		}
   191  		return val;
   192  	}
   193  	
   194  	private function setColor (obj:Object, hex:Number):Void {
   195  		var c = new Color (obj);
   196  		c.setRGB (hex);
   197  	}
   198  	
   199  	// MOUSE WHEEL HANDLER
   200  	
   201  	private function setMouseWheel ():Void {
   202  		if (_isMouseWheelEnabled) {
   203  			//deactivate mousewheel on textfield & activate mouse listener
   204  			_content.mouseWheelEnabled = false;
   205  			Mouse.addListener (_this);
   206  		}
   207  	}
   208  	
   209  	private function isMouseInArea () {
   210  		var x:Number = _content._parent._xmouse;
   211  		var y:Number = _content._parent._ymouse;
   212  		return (x >= 0 && x <= _mask._width && y >= 0 && y <= _mask[_prop]);
   213  	}
   214  	
   215  	private function onMouseWheel (delta:Number) {
   216  		if (isMouseInArea () && isScrollable ()) {
   217  			var m:Number = _scroller[_axis] - ((_gutter[_prop] / 5) * (delta / 3));
   218  			slideScroller (m);
   219  		}
   220  	}
   221  	
   222  	// UI EVENT HANDLERS
   223  	
   224  	private function scroller_onEnterFrame () {
   225  		_scroller._visible = isScrollable ();
   226  		// height changed, adjust scroller
   227  		if (_contentBot != _contentTop + _contentSizeTracker[_prop]) {
   228  			_contentBot = _contentTop + _contentSizeTracker[_prop];
   229  			updateConfines ();
   230  			_scroller[_axis] = resolveScrollerPos ((_contentTop + _content[_axis]) * (_gutterBot - _gutterTop) / (_gutterBot - _contentBot + _scroller[_prop]) + _gutterTop);
   231  			//updateConfines (); // !!! experimental !!! - should update proportional scroller and associated vars
   232  		}
   233  		//                                                                                                   
   234  		if (!_isDragged) {
   235  			_oldPos = _scroller[_axis];
   236  			_newPos = _oldPos + _speed;
   237  			if (_newPos > _gutterBot || _newPos < _gutterTop) {
   238  				//bounce - reverse movement
   239  				_speed = -_speed;
   240  				_newPos = _oldPos;
   241  			}
   242  			_speed = Math.round (_speed * _friction * 100) / 100;
   243  		} else {
   244  			_oldPos = _newPos;
   245  			_newPos = _scroller[_axis];
   246  		}
   247  		//update scroller
   248  		_scroller[_axis] = _newPos;
   249  		//update content
   250  		var percent:Number = (_scroller[_axis] - _gutterTop) / (_gutterBot - _gutterTop);
   251  		_content[_axis] = Math.round ((percent * (_gutterBot - _contentBot + _scroller[_prop])) + _contentTop);
   252  		//weird throw bug fix - gets stuck on speed 0.05 when scroller is thrown upwards and decelerates
   253  		if (_speed < 0.1 && _speed > -0.1) {
   254  			_speed = 0;
   255  		}
   256  	}
   257  	
   258  	private function scroller_onRollOver () {
   259  		if (!isNaN (_colors.over)) {
   260  			setColor (_scroller,_colors.over);
   261  		}
   262  	}
   263  	
   264  	private function scroller_onRollOut () {
   265  		if (!isNaN (_colors.up)) {
   266  			setColor (_scroller,_colors.up);
   267  		}
   268  	}
   269  	
   270  	private function scroller_onPress () {
   271  		stopScroller ();
   272  		_isDragged = true;
   273  		if(_axis == "_y") {
   274  			_scroller.startDrag (false, _scroller._x, _gutterTop, _scroller._x, _gutterBot);
   275  		} else {
   276  			_scroller.startDrag (false, _gutterTop, _scroller._y, _gutterBot, _scroller._y);
   277  		}
   278  	}
   279  	
   280  	private function scroller_onRelease () {
   281  		_scroller.stopDrag ();
   282  		_isDragged = false;
   283  		//throw
   284  		_speed = (_newPos - _oldPos) * _ratio;
   285  	}
   286  	
   287  	private function scroller_onReleaseOutside () {
   288  		scroller_onRelease ();
   289  		scroller_onRollOut ();
   290  	}
   291  	
   292  	private function gutter_onPress () {
   293  		if (isScrollable ()) {
   294  			_gutter.useHandCursor = true;
   295  			var mousePos:Number = (_axis == "_y") ? _gutter._parent._ymouse : _gutter._parent._xmouse;
   296  			slideScroller (mousePos - (_scroller[_prop] / 2));
   297  		} else {
   298  			_gutter.useHandCursor = false;
   299  		}
   300  	}
