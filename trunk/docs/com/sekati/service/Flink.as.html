     8  import flash.external.ExternalInterface;
     9  /**
    10   * Easily add deeplinking into flash sites
    11   * 
    12   * {@code Usage:
    13   * // Add Flink.js to your swf html page:
    14   * <script language="javascript" src="js/Flink.js"></script>
    15   *
    16   * // enable swLiveConnect:
    17   * fscommand ("swLiveConnect", "true");
    18   * }
    19   */
    20  class com.sekati.service.Flink {
    21  	private static var _link:String;
    22  	private static var _isAvailable:Boolean = ExternalInterface.available;
    23  	/**
    24  	 * get deeplink anchor
    25  	 * @return String
    26  	 * {@code Usage:
    27  	 * var currentDeeplink:String = Flink.getLink();
    28  	 * }
    29  	 */
    30  	public static function getLink ():String {
    31  		_link = String (ExternalInterface.call ("getFlink"));
    32  		trace ("*** Flink.getLink = '" + _link + "'");
    33  		return (_link);
    34  	}
    35  	/**
    36  	 * set deeplink anchor and page title
    37  	 * @return String
    38  	 * {@code
    39  	 * Flink.setLink("Section Three","section_three");
    40  	 * }
    41  	 */
    42  	public static function setLink (pgTitle:String, anchor:String):Void {
    43  		ExternalInterface.call ("setFlink", pgTitle, anchor);
    44  		_link = anchor;
    45  		trace ("*** Flink.setLink = '" + _link + "'");
    46  	}
    47  	/**
    48  	 * call a javascript function from actionscript
    49  	 * @return Void
    50  	 * {@code
    51  	 *  Flink.jsCall ("helloWorld", "boy", "jason");
    52  	 * }
    53  	 */
    54  	public static function jsCall () {
    55  		ExternalInterface.call.apply (null, arguments);
    56  	}
    57  	/**
    58  	 * allow javascript access to call an actionscript function
    59  	 * @param fnName (String) as function name as string
    60  	 * @param instance (Object) scope of fnName
    61  	 * @param fn (Function) as function name
    62  	 * @return Void
    63  	 * {@code Usage:
    64  	 * Flink.jsAllow ("helloFlash", this, helloFlash);
    65  	 * }
    66  	 */
    67  	public static function jsAllow (fnName:String, instance:Object, fn:Function):Void {
    68  		trace ("*** Flink.setAvailable: '" + fnName + "'");
    69  		ExternalInterface.addCallback (fnName, instance, fn);
    70  	}
    71  	//
    72  }
    73  // eof
