    11  /**
    12   * Encodes and decodes a base8 (hex) string.
    13   */
    14  class com.sekati.crypt.Base8 {
    15  
    16  	/**
    17  	* Encodes a base8 string.
    18  	* @param src (String) - string to encode
    19  	* @return String 
    20      */
    21  	public static function encode(src:String):String {
    22  		var result:String = new String("");
    23  		for (var i:Number = 0; i<src.length; i++) {
    24  			result += src.charCodeAt(i).toString(16);
    25  		}
    26  		return result;
    27  	}
    28  
    29  	/**
    30  	* Decodes a base8 string.
    31  	* @param src (String) - string to decode
    32  	* @return String 
    33  	*/
    34  	public static function decode(src:String):String {
    35  		var result:String = new String("");
    36  		for (var i:Number = 0; i<src.length; i+=2) {
    37  			result += String.fromCharCode(parseInt(src.substr(i, 2), 16));
    38  		}
    39  		return result;
    40  	}

    42  }
