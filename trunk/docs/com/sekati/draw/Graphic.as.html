   104  		_mc.lineStyle (thick,color);

   112  	// draw Oval - improved!
   113  	function drawOval2 (x:Number,y:Number,width:Number,height:Number) {
   114  		var j = width * 0.70711;
   115  		var n = height * 0.70711;
   116  		var i = j - (height - n) * width / height;
   117  		var m = n - (width - j) * height / width;
   118  		_mc.lineStyle(thick, color);
   119  		_mc.moveTo(x + width, y);
   120  		_mc.curveTo(x + width, y - m, x + j, y - n);
   121  		_mc.curveTo(x + i, y - height, x, y - height);
   122  		_mc.curveTo(x - i, y - height, x - j, y - n);
   123  		_mc.curveTo(x - width, y - m, x - width, y);
   124  		_mc.curveTo(x - width, y + m, x - j, y + n);
   125  		_mc.curveTo(x - i, y + height, x, y + height);
   126  		_mc.curveTo(x + i, y + height, x + j, y + n);
   127  		_mc.curveTo(x + width, y + m, x + width, y);
   128  	}

   142  	//---------DRAW CIRCLE-----------//
   143  	//if argument styleMaker == 22.5 you get a full circle 
   144  	//  66 makes a star like figure
   145  	// usage :  drawCircle(radius,x,y,22.5)
   146  	function drawCircle (r:Number, x:Number, y:Number) {
   147  		var styleMaker:Number = 22.5;
   148  		_mc.moveTo (x + r,y);
   149  		_mc.lineStyle (thick,color);
   150  		var style:Number = Math.tan (styleMaker * Math.PI / 180);
   151  		for (var angle:Number = 45; angle <= 360; angle += 45) {
   152  			var endX:Number = r * Math.cos (angle * Math.PI / 180);
   153  			var endY:Number = r * Math.sin (angle * Math.PI / 180);
   154  			var cX:Number = endX + r * style * Math.cos ((angle - 90) * Math.PI / 180);
   155  			var cY:Number = endY + r * style * Math.sin ((angle - 90) * Math.PI / 180);
   156  			_mc.curveTo (cX + x,cY + y,endX + x,endY + y);
   157  		}
   158  	}

   261  	public function drawHexagon (hexRadius:Number, startX, startY) {
   262  		var sideC:Number = hexRadius;
   263  		var sideA:Number = 0.5 * sideC;
   264  		var sideB:Number = Math.sqrt ((hexRadius * hexRadius) - (0.5 * hexRadius) * (0.5 * hexRadius));
   265  		_mc.lineStyle (thick,color,100);
   266  		_mc.moveTo (startX,startY);
   267  		_mc.lineTo (startX,sideC + startY);
   268  		_mc.lineTo (sideB + startX,startY + sideA + sideC);
   269  		// bottom point
   270  		_mc.lineTo (2 * sideB + startX,startY + sideC);
   271  		_mc.lineTo (2 * sideB + startX,startY);
   272  		_mc.lineTo (sideB + startX,startY - sideA);
   273  		_mc.lineTo (startX,startY);
   274  	}
