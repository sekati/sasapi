    70  	/**
    71  	 * Splits the _color parameter into specific color variables
    72  	 *
    73  	 * @param		p_value				Number		The original _color value
    74  	 * @return							Array		An array containing the .name and .value of all new properties
    75  	 */
    76  	public static function _color_splitter (p_value:Number):Array {
    77  		var nArray:Array = new Array();
    78  		if (p_value == null) {
    79  			// No parameter passed, so just resets the color
    80  			nArray.push({name:"_color_ra", value:100});
    81  			nArray.push({name:"_color_rb", value:0});
    82  			nArray.push({name:"_color_ga", value:100});
    83  			nArray.push({name:"_color_gb", value:0});
    84  			nArray.push({name:"_color_ba", value:100});
    85  			nArray.push({name:"_color_bb", value:0});
    86  		} else {
    87  			// A color tinting is passed, so converts it to the object values
    88  			nArray.push({name:"_color_ra", value:0});
    89  			nArray.push({name:"_color_rb", value:AuxFunctions.numberToR(p_value)});
    90  			nArray.push({name:"_color_ga", value:0});
    91  			nArray.push({name:"_color_gb", value:AuxFunctions.numberToG(p_value)});
    92  			nArray.push({name:"_color_ba", value:0});
    93  			nArray.push({name:"_color_bb", value:AuxFunctions.numberToB(p_value)});
    94  		}
    95  		return nArray;
    96  	}

   109  		var nArray:Array = new Array();
   110  		if (p_value == null) {
   111  			// No parameter passed, so just resets the color
   112  			nArray.push({name:"_color_ra", value:100});
   113  			nArray.push({name:"_color_rb", value:0});
   114  			nArray.push({name:"_color_ga", value:100});
   115  			nArray.push({name:"_color_gb", value:0});
   116  			nArray.push({name:"_color_ba", value:100});
   117  			nArray.push({name:"_color_bb", value:0});
   118  		} else {
   119  			// A color tinting is passed, so converts it to the object values
   120  			if (p_value.ra != undefined) nArray.push({name:"_color_ra", value:p_value.ra});
   121  			if (p_value.rb != undefined) nArray.push({name:"_color_rb", value:p_value.rb});
   122  			if (p_value.ga != undefined) nArray.push({name:"_color_ba", value:p_value.ba});
   123  			if (p_value.gb != undefined) nArray.push({name:"_color_bb", value:p_value.bb});
   124  			if (p_value.ba != undefined) nArray.push({name:"_color_ga", value:p_value.ga});
   125  			if (p_value.bb != undefined) nArray.push({name:"_color_gb", value:p_value.gb});
   126  			if (p_value.aa != undefined) nArray.push({name:"_color_aa", value:p_value.aa});
   127  			if (p_value.ab != undefined) nArray.push({name:"_color_ab", value:p_value.ab});
   128  		}
   129  		return nArray;
   130  	}

   287  	/**
   288  	 * (filters)
   289  	 * Generic function for the properties of filter objects
   290  	 */
   291  	public static function _filter_property_get (p_obj:Object, p_parameters:Array):Number {
   292  		var f:Array = p_obj.filters;
   293  		var i:Number;
   294  		var filterClass:Object = p_parameters[0];
   295  		var propertyName:String = p_parameters[1];
   296  		for (i = 0; i < f.length; i++) {
   297  			if (f[i] instanceof filterClass) return (f[i][propertyName]);
   298  		}
   299  
   300  		// No value found for this property - no filter instance found using this class!
   301  		// Must return default desired values
   302  		var defaultValues:Object;
   303  		switch (filterClass) {
   304  			case BlurFilter:
   305  				defaultValues = {blurX:0, blurY:0, quality:NaN};
   306  				break;
   307  		}
   308  		// When returning NaN, the Tweener engine sets the starting value as being the same as the final value
   309  		// When returning null, the Tweener engine doesn't tween it at all, just setting it to the final value
   310  		return defaultValues[propertyName];
   311  	}
   312  
   313  	public static function _filter_property_set (p_obj:Object, p_value:Number, p_parameters:Array):Void {
   314  		var f:Array = p_obj.filters;
   315  		var i:Number;
   316  		var filterClass:Object = p_parameters[0];
   317  		var propertyName:String = p_parameters[1];
   318  		for (i = 0; i < f.length; i++) {
   319  			if (f[i] instanceof filterClass) {
   320  				f[i][propertyName] = p_value;
   321  				p_obj.filters = f;
   322  				return;
   323  			}
   324  		}
   325  
   326  		// The correct filter class wasn't found - create a new one
   327  		if (f == undefined) f = new Array();
   328  		var fi:BitmapFilter;
   329  		switch (filterClass) {
   330  			case BlurFilter:
   331  				fi = new BlurFilter(0, 0);
   332  				break;
   333  		}
   334  		fi[propertyName] = p_value;
   335  		f.push(fi);
   336  		p_obj.filters = f;
   337  	}

   347  	/**
   348  	 * Given the parameter object passed to this special property, return an array listing the properties that should be modified, and their parameters
   349  	 *
   350  	 * @param		p_obj				Object		Parameter passed to this property
   351  	 * @return							Array		Array listing name and parameter of each property
   352  	 */
   353  	public static function _bezier_modifier (p_obj:Object):Array {
   354  		var mList:Array = []; // List of properties to be modified
   355  		var pList:Array; // List of parameters passed, normalized as an array
   356  		if (p_obj instanceof Array) {
   357  			// Complex
   358  			pList = p_obj.concat();
   359  		} else {
   360  			pList = [p_obj];
   361  		}
   362  
   363  		var i:Number;
   364  		var istr:String;
   365  		var mListObj:Object = {}; // Object describing each property name and parameter
   366  
   367  		for (i = 0; i < pList.length; i++) {
   368  			for (istr in pList[i]) {
   369  				if (mListObj[istr] == undefined) mListObj[istr] = [];
   370  				mListObj[istr].push(pList[i][istr]);
   371  			}
   372  		}
   373  		for (istr in mListObj) {
   374  			mList.push({name:istr, parameters:mListObj[istr]});
   375  		}
   376  		return mList;
   377  	}

   379  	/**
   380  	 * Given tweening specifications (beging, end, t), applies the property parameter to it, returning new t
   381  	 *
   382  	 * @param		b					Number		Beginning value of the property
   383  	 * @param		e					Number		Ending (desired) value of the property
   384  	 * @param		t					Number		Current t of this tweening (0-1), after applying the easing equation
   385  	 * @param		p					Array		Array of parameters passed to this specific property
   386  	 * @return							Number		New t, with the p parameters applied to it
   387  	 */
   388  	public static function _bezier_get (b:Number, e:Number, t:Number, p:Array):Number {
   389  		// This is based on Robert Penner's code
   390  		if (p.length == 1) {
   391  			// Simple curve with just one bezier control point
   392  			return b + t*(2*(1-t)*(p[0]-b) + t*(e - b));
   393  		} else {
   394  			// Array of bezier control points, must find the point between each pair of bezier points
   395  			var ip:Number = Math.floor(t * p.length); // Position on the bezier list
   396  			var it:Number = (t - (ip * (1 / p.length))) * p.length; // t inside this ip
   397  			var p1:Number, p2:Number;
   398  			if (ip == 0) {
   399  				// First part: belongs to the first control point, find second midpoint
   400  				p1 = b;
   401  				p2 = (p[0]+p[1])/2;
   402  			} else if (ip == p.length - 1) {
   403  				// Last part: belongs to the last control point, find first midpoint
   404  				p1 = (p[ip-1]+p[ip])/2;
   405  				p2 = e;
   406  			} else {
   407  				// Any middle part: find both midpoints
   408  				p1 = (p[ip-1]+p[ip])/2;
   409  				p2 = (p[ip]+p[ip+1])/2;
   410  			}
   411  			return p1+it*(2*(1-it)*(p[ip]-p1) + it*(p2 - p1));
   412  		}
   413  	}
